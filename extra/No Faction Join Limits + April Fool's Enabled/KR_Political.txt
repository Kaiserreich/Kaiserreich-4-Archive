country_event = { #Very Good Stability -> Normal Stability
	id = kr.political.1
	title = kr.political.1.t
	desc = kr.political.1.d
	picture = GFX_report_event_generic_rally_3

	trigger = {
		has_political_power < 0
		has_idea = very_good_stability
	}

	mean_time_to_happen = {
		days = 50

		modifier = {
			factor = 0.75
			has_political_power < -150
			has_political_power > -299
		}
		modifier = {
			factor = 0.1
			has_political_power < -300
		}
	}

	option = {
		name = kr.political.1.a
		swap_ideas = {
			remove_idea = very_good_stability
			add_idea = normal_stability
		}
		set_country_flag = normal_stability
	}
}

country_event = {
	id = kr.political.11
	title = kr.political.11.t
	desc = kr.political.11.d
	picture = popularity_rising

	fire_only_once = no

	trigger = {
		NOT = { has_idea = very_high_popularity }
		OR = {
			AND = {
				has_government = totalist
				totalist > 0.80
			}
			AND = {
				has_government = radical_socialist
				radical_socialist > 0.80
			}
			AND = {
				has_government = syndicalist
				syndicalist > 0.80
			}
			AND = {
				has_government = social_democrat
				social_democrat > 0.80
			}
			AND = {
				has_government = social_liberal
				social_liberal > 0.80
			}
			AND = {
				has_government = market_liberal
				market_liberal > 0.80
			}
			AND = {
				has_government = social_conservative
				social_conservative > 0.80
			}
			AND = {
				has_government = authoritarian_democrat
				authoritarian_democrat > 0.80
			}
			AND = {
				has_government = paternal_autocrat
				paternal_autocrat > 0.80
			}
			AND = {
				has_government = national_populist
				national_populist > 0.80
			}
		}
	}

	mean_time_to_happen = {
		days = 2
	}

	option = {
		name = kr.political.11.a
		if = {
			limit = { has_idea = very_low_popularity }
			swap_ideas = {
				remove_idea = very_low_popularity
				add_idea = very_high_popularity
			}
		}
		if = {
			limit = { has_idea = low_popularity }
			swap_ideas = {
				remove_idea = low_popularity
				add_idea = very_high_popularity
			}
		}
		if = {
			limit = { has_idea = normal_popularity }
			swap_ideas = {
				remove_idea = normal_popularity
				add_idea = very_high_popularity
			}
		}
		if = {
			limit = { has_idea = high_popularity }
			swap_ideas = {
				remove_idea = high_popularity
				add_idea = very_high_popularity
			}
		}
	}

}
