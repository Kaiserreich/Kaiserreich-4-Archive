# Scripted AI desire for peace actions.
# See documentation.info file for more detailed scripting information.

peace_ai_desires = {
	base_puppet_liberate_force_government_desire = {
		peace_action_type = { puppet liberate force_government }
		enable = {
			always = yes
		}
		ai_desire = -400
	}

	ignore_puppet_liberate_or_force_government_if_claimed = {
		peace_action_type = { puppet liberate force_government }
		enable = {
			NOT = { tag = ROOT.FROM }
			ROOT.FROM.FROM.FROM = {
				OR = {
					is_claimed_by = ROOT
					is_core_of = ROOT
					has_variable = ambition_area_@ROOT
				}
			}
		}
		ai_desire = -1000
	}

	ignore_puppet_liberate_or_force_government_if_claimed_by_ally = {
		peace_action_type = { puppet liberate force_government }
		enable = {
			is_in_faction = yes
			any_allied_country = {
				pc_is_winner = yes
				if = {
					limit = { is_subject = yes }
					ROOT.FROM.FROM.FROM = {
						OR = {
							is_claimed_by = PREV
							is_core_of = PREV
						}
					}
				}
				else = {
					ROOT.FROM.FROM.FROM = {
						OR = {
							is_claimed_by = PREV
							is_core_of = PREV
							has_variable = ambition_area_@PREV
						}
					}
				}
			}
		}
		ai_desire = -1000
	}

	dont_revive_colonial_states = {
		peace_action_type = { puppet liberate force_government }
		enable = {
			ROOT.FROM = {
				OR = {
					original_tag = MAF
					original_tag = GEA
					original_tag = DEI
				}
			}
		}
		ai_desire = -1000
	}

	liberate_countries_in_africa = {
		peace_action_type = { liberate }
		enable = {
			is_in_africa = no
			ROOT.FROM.FROM.FROM = { is_on_continent = africa }
			ROOT.FROM = {
				is_in_africa = yes
				exists = no
				NOT = { tag = ROOT }
				is_in_north_africa = no
				NOT = { original_tag = SAF }
			}
		}
		ai_desire = 400 #offsets do_not_puppet_liberate_or_force_government
	}

	force_government_in_africa_if_already_exists = {
		peace_action_type = { force_government }
		enable = {
			is_in_africa = no
			ROOT.FROM.FROM.FROM = { is_on_continent = africa }
			ROOT.FROM = {
				is_in_africa = yes
				exists = yes
				NOT = { tag = ROOT }
				is_in_north_africa = no
				NOT = { original_tag = SAF }
			}
		}
		ai_desire = 400 #offsets do_not_puppet_liberate_or_force_government
	}
}
