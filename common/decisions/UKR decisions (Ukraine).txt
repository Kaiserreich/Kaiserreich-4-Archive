	UKR_evacuate_industry = { #should this be made dynamic and repeatable?
		icon = GFX_decision_factories
		allowed = { always = no }

		available = {
			threat > 0.5
			any_owned_state = {
				is_owned_and_controlled_by = ROOT
				any_neighbor_state = { is_controlled_by = RUS } #Moscow
				is_in_array = { UKR.UKR_states_east_of_dnieper_array = THIS }
				arms_factory > 0
			}
			custom_trigger_tooltip = {
				tooltip = UKR_holds_east_dnipro
				any_of_scopes = {
					array = UKR.UKR_states_east_of_dnieper_array
					is_controlled_by = ROOT
					is_in_home_area = yes
				}
			}
		}
		visible = {
			is_neighbor_of = RUS
			RUS = {
				is_subject = no
				has_capitulated = no
				NOT = { is_in_faction_with = ROOT }
			}
		}

		days_remove = 30
		fire_only_once = no
		cost = UKR_evacuation_cost
		ai_will_do = { factor = 100 }

		complete_effect = {
			log = "[GetLogRoot]: Decision complete UKR_evacuate_industry"
			add_to_variable = { UKR_evacuation_cost = 5 }
			random_owned_controlled_state = {
				limit = {
					any_neighbor_state = { is_owned_and_controlled_by = RUS } #Moscow
					arms_factory > 0
				}
				remove_building = {
					type = arms_factory
					level = 1
				}
			}
		}

		remove_effect = {
			log = "[GetLogRoot]: Decision remove UKR_evacuate_industry"
			random_owned_controlled_state = {
				limit = {
					OR = {
						state = 201
						state = 202
						state = 203
						state = 192
						state = 197
						state = 198
						state = 199
						state = 792
						state = 93
					}
					free_building_slots = { building = arms_factory size > 0 include_locked = yes }
				}
				add_building_construction = {
					type = arms_factory
					level = 1
					instant_build = yes
				}
			}
		}
