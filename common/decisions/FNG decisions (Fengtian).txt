	FNG_Join_The_Zhifeng_War = {
		icon = generic_prepare_civil_war

		allowed = { always = no }

		visible = {
			OR = {

			}

			}
		}

		fire_only_once = yes
		cancel_if_not_visible = yes

		cancel_effect = {
			log = "[GetLogRoot][GetLogFrom]: Decision cancel FNG_Join_The_Zhifeng_War"
			clear_decision_attack_AI = yes
		}

		complete_effect = {
			log = "[GetLogRoot][GetLogFrom]: Decision complete FNG_Join_The_Zhifeng_War"
			set_country_flag = FNG_anguojun_target
			setup_decision_attack_AI = yes
			hidden_effect = {
				if = {
					limit = {
						FROM = {
							decryption_progress = {
								target = ROOT
								value > 0.5
							}
						}
					}
					FROM = {
						country_event = { id = fengtian_events.255 days = 7 }	#warning event
						if = {
							limit = {
								is_subject = yes
							}
							overlord = {
								country_event = { id = fengtian_events.256 days = 7 }
							}
						}
					}
				}
				if = {
					limit = {
						OR = {
							AND = {
								NOT = {
									has_tech = basic_encryption
								}
								FROM = {
									has_tech = basic_decryption
								}
							}
							AND = {
								NOT = {
									has_tech = improved_encryption
								}
								FROM = {
									has_tech = improved_decryption
								}
							}
							AND = {
								NOT = {
									has_tech = advanced_encryption
								}
								FROM = {
									has_tech = advanced_decryption
								}
							}
						}
					}
					FROM = {
						country_event = { id = fengtian_events.255 days = 7 }
						if = {
							limit = {
								is_subject = yes
							}
							overlord = {
								country_event = { id = fengtian_events.256 days = 7 }
							}
						}
					}
				}
				else = {
					FROM = {
						country_event = { id = fengtian_events.255 days = 13 }
						if = {
							limit = {
								is_subject = yes
							}
							overlord = {
								country_event = { id = fengtian_events.256 days = 13 }
							}
						}
					}
				}
			}
		}
		remove_effect = {
			log = "[GetLogRoot][GetLogFrom]: Decision remove FNG_Join_The_Zhifeng_War"
			clr_country_flag = FNG_anguojun_target
			clear_decision_attack_AI = yes
			declare_war_on = {
				target = FROM
				type = annex_everything
			}
			give_military_access = FNG
			FNG = { give_military_access = ROOT }
			every_other_country = {
				limit = {
					is_subject_of = ROOT
				}
				give_military_access = FNG
			}
		}
		fixed_random_seed = no
	}
