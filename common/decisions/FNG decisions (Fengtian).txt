			}
		}

		cancel_if_not_visible = yes

		ai_will_do = { factor = 1000 }

		cancel_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision cancel FNG_Join_The_Zhifeng_War target: [From.GetName]"
			clear_decision_attack_AI = yes
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete FNG_Join_The_Zhifeng_War target: [From.GetName]"
			set_country_flag = FNG_anguojun_target
			setup_decision_attack_AI = yes
			hidden_effect = {
				if = {
					limit = {
						FROM = {
							decryption_progress = {
								target = ROOT
								value > 0.5
							}
						}
					}
					FROM = {
						country_event = { id = fengtian.255 days = 7 }	#warning event
						if = {
							limit = {
								is_subject = yes
							}
							overlord = {
								country_event = { id = fengtian.256 days = 7 }
							}
						}
					}
				}
				if = {
					limit = {
						OR = {
							AND = {
								NOT = {
									has_tech = basic_encryption
								}
								FROM = {
									has_tech = basic_decryption
								}
							}
							AND = {
								NOT = {
									has_tech = improved_encryption
								}
								FROM = {
									has_tech = improved_decryption
								}
							}
							AND = {
								NOT = {
									has_tech = advanced_encryption
								}
								FROM = {
									has_tech = advanced_decryption
								}
							}
						}
					}
					FROM = {
						country_event = { id = fengtian.255 days = 7 }
						if = {
							limit = {
								is_subject = yes
							}
							overlord = {
								country_event = { id = fengtian.256 days = 7 }
							}
						}
					}
				}
				else = {
					FROM = {
						country_event = { id = fengtian.255 days = 13 }
						if = {
							limit = {
								is_subject = yes
							}
							overlord = {
								country_event = { id = fengtian.256 days = 13 }
							}
						}
					}
				}
			}
		}
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove FNG_Join_The_Zhifeng_War target: [From.GetName]"
			clr_country_flag = FNG_anguojun_target
			clear_decision_attack_AI = yes

			}
