	YUN_troops_flood_to_the_border_monthly_timer = {
		allowed = {
			always = no
		}
		available = {
			hidden_trigger = { always = no }
		}
		icon = GFX_decision_generic_prepare_civil_war
		is_good = yes
		priority = 90 #Should be at the top
		days_mission_timeout = 30
		complete_effect = {
			log = "[GetLogRoot]: Decision complete YUN_troops_flood_to_the_border_monthly_timer"
		}
		timeout_effect = {
			log = "[GetLogRoot]: Decision timeout YUN_troops_flood_to_the_border_monthly_timer"
			#NOTE: REMEMBER SINCE THE COUNTER IS ADDED LAST, REFLECTS CURRENT MONTH STARTING WITH 1 (SET BY YUN.INTROGAME.006)
			#Boost the Losing Side
			custom_effect_tooltip = YUN_troops_flood_to_the_border_monthly_timer_tt
			if = {
				limit = { check_variable = { YUN_introminigame_KMT_soldiers_var > YUN_introminigame_UPC_soldiers_var } }
				YUN_introminigame_add_UPC_small_effect = yes
			}
			else_if = {
				limit = { check_variable = { YUN_introminigame_UPC_soldiers_var > YUN_introminigame_KMT_soldiers_var } }
				YUN_introminigame_add_KMT_small_effect = yes
			}
			else = {
				random_list = {
					50 = {
						YUN_introminigame_add_KMT_small_effect = yes
					}
					50 = {
						YUN_introminigame_add_UPC_small_effect = yes
					}
				}
			}

			#Now add to the month counter variable
			if = {
				limit = { check_variable = { YUN.YUN_introminigame_month_counter_var < 8 } }
				#Keep repeating this until November 1st.
				activate_mission = YUN_troops_flood_to_the_border_monthly_timer
				add_to_variable = { YUN.YUN_introminigame_month_counter_var = 1 }
			}
			else = {
				custom_effect_tooltip = YUN_introminigame_winter_offensive_tt
				country_event = YUN.introgame.100 #The November Offensive - Basically the latest possible time you can delay choosing a path.
			}
		}
		fixed_random_seed = no
	}
