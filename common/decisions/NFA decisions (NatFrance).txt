			NOT = { country_exists = GER }

			can_declare_war_on = FROM

				OR = {
					is_subject = no
					is_subject_of = GEX
				}
			}
		}

		ai_will_do = {
			factor = 1000
			modifier = {
				factor = 0
				country_is_in_dangerous_war = yes
			}
			modifier = {
				factor = 0
				FROM = { default_attack_AI_trigger = yes }
			}
			modifier = {
				factor = 0
				FROM = { dont_start_major_faction_wars = yes }
			}
		}

		cancel_effect = {
			log = "[GetLogRoot][GetLogFrom]: Decision cancel NFA_seize_real_morocco"
			clear_decision_attack_AI = yes
		}

		complete_effect = {
			log = "[GetLogRoot][GetLogFrom]: Decision complete NFA_seize_real_morocco"
			FROM = { incoming_war_notification_effect = yes }
			setup_decision_attack_AI = yes
		}

		remove_effect = {
			log = "[GetLogRoot][GetLogFrom]: Decision remove NFA_seize_real_morocco"
			clear_decision_attack_AI = yes
			if = {
				limit = {
					FROM = { is_subject_of = GEX }
				}

					country_event = { id = nfa.111 days = 2 }
				}
				add_opinion_modifier = {
					target = GEX
					modifier = NFA_seized_morocco
				}
			}
			else = {

					country_event = { id = nfa.116 days = 2 }

			}
