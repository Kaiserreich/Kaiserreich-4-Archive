					}
				}
			}
		}
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove ANQ_road_reconstruction_decision"

				limit = {

				}
				596 = {

						type = infrastructure

						instant_build = yes
					}
				}
			}
