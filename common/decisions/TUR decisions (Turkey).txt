TUR_deal_with_OHF = {
	TUR_deal_with_OHF = {
		icon = GFX_decision_generic_political_discourse

		allowed = { always = no }

		available = {
			controls_state = FROM
		}
		cancel_trigger = {
			FROM = {
				OR = {
					NOT = { is_owned_and_controlled_by = ROOT }
					NOT = { has_dynamic_modifier = { modifier = TUR_centralist_control_modifier } }
				}
			}
		}

		cost = 35
		days_remove = 40
		fire_only_once = yes
		state_target = yes
		on_map_mode = map_and_decisions_view

		ai_will_do = {
			factor = 50
			modifier = {
				factor = 2
				has_political_power > 70
			}
		}

		cancel_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision cancel TUR_deal_with_OHF target: [From.GetName]"
			hidden_effect = {
				add_political_power = 35
				if = {
					limit = {
						OR = {
							NOT = { owns_state = FROM }
							NOT = { FROM = { has_dynamic_modifier = { modifier = TUR_centralist_control_modifier } } }
						}
					}
					remove_targeted_decision = { decision = TUR_deal_with_OHF target = FROM }
				}
			}
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove TUR_deal_with_OHF target: [From.GetName]"
			FROM = {
				clr_state_flag = TUR_centralist_control_state
				remove_dynamic_modifier = { modifier = TUR_centralist_control_modifier }
			}
			hidden_effect = { remove_targeted_decision = { decision = TUR_deal_with_OHF target = FROM } }
		}
	}
}
