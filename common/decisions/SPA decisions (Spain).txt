foreign_politics = {
	SPA_join_faction = {
		icon = GFX_decision_hol_exchange_intelligence_data

		allowed = { always = no } #Manually activated by foreign policy foci

		available = {
			at_peace_or_same_enemies_as_FROM = yes
		}

		visible = {
			is_subject = no
			is_in_faction = no
			if = {
				limit = {
					FROM = { tag = ENT }
					ROOT = { has_completed_focus = SPA_franco_spain }
				}
				NOT = { country_exists = NFA }
			}
			NOT = { has_country_flag = SPA_joining_faction }
		}

		targets = { GER ENT INT }
		target_root_trigger = {
			is_subject = no
			is_in_faction = no
		}
		target_trigger = {
			FROM = { is_faction_leader = yes }
			NOT = { has_war_with = FROM }
		}
		cost = 20
		fire_only_once = yes

		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				FROM = { tag = GER }
				ROOT = { tag = SPK }
				has_game_rule = { rule = SPA_forpol option = SPA_FORPOL_ENT }
			}
			modifier = {
				factor = 0
				FROM = { tag = ENT }
				ROOT = { tag = SPK }
				has_game_rule = { rule = SPA_forpol option = SPA_FORPOL_GER }
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete SPA_join_faction target: [From.GetName]"
			FROM = { country_event = sprd.31 }
			custom_effect_tooltip = if_they_accept_tt
			effect_tooltip = { FROM = { add_to_faction = ROOT } }
			if = {
				limit = { FROM = { tag = ENT } has_completed_focus = SPA_bourbon_throne }
				custom_effect_tooltip = SPA_remove_claims_on_FRA_tt
			}
			set_country_flag = SPA_joining_faction
		}
	}
}
