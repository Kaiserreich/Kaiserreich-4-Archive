ICE_Fishing_Regions_decisions = {
	ICE_Greenland_Region = {
		icon = GFX_decision_generic_naval
		allowed = { always = no }			# Activated by ICE_Expand_Fishing_Regions

		visible = {
			NOT = { has_country_flag = ICE_healthy_economy_cf }
		}

		available = {
			OR = {
				NOT = {
					is_subject_of = DEN
				}
				AND = {
					is_subject_of = DEN
					has_country_flag = DEN_allowed_greenland_fishing
				}
			}
		}

		ai_will_do = { factor = 20 }

		fire_only_once = yes
		cost = 30
		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete ICE_Greenland_Region"
			hidden_effect = {
				set_country_flag = ICE_Greenland_Fishing
			}
			if = {
				limit = {
					has_idea = ICE_recovering_economy_4
				}
				swap_ideas = {
					remove_idea = ICE_recovering_economy_4
					add_idea = ICE_recovering_economy_5
				}
			}
			else_if = {
				limit = {
					has_idea = ICE_recovering_economy_6
				}
				swap_ideas = {
					remove_idea = ICE_recovering_economy_6
					add_idea = ICE_healthy_economy_1
				}
				add_stability = 0.04
				hidden_effect = {
					ICE = {
						set_country_flag = ICE_healthy_economy_cf
						country_event = { id = ice.1 days = 5 }
					}
				}
			}
		}
	}
