political_actions = {
	GER_appeal_to_old_friends = {
		icon = generic_civil_support

		allowed = {
			original_tag = GER
		}

		available = {
			FROM = {
				controls_all_owned_states = yes
				at_peace_or_same_enemies_as_ROOT = yes
			}
		}

		visible = {
			surrender_progress > 0.25
			is_subject = no
			has_socialist_government = no
			NOT = {
				is_in_faction_with = FROM
				has_war_with = FROM
			}
			FROM = {
				exists = yes
				NOT = { has_country_flag = GER_appealed_old_friends }
				is_subject = no
				has_socialist_government = no
				OR = {
					is_in_faction = no
					is_faction_leader = yes
				}
			}
		}

		target_root_trigger = {
			surrender_progress > 0.35
		}

		targets = { AUS BUL }

		target_trigger = {
			FROM = {
				NOT = { has_country_flag = GER_appealed_old_friends }
				is_subject = no
				has_socialist_government = no
				NOT = { has_government = national_populist }
				OR = {
					is_in_faction = no
					is_faction_leader = yes
				}
			}
		}

		fire_only_once = no

		cost = 0

		ai_will_do = {
			factor = 100
			modifier = {
				factor = 0
				FROM = {
					has_war = yes
					any_enemy_country = {
						NOT = { has_war_with = ROOT }
					}
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete GER_appeal_to_old_friends target: [From.GetName]"
			if = {
				limit = { FROM = { tag = AUS } }
				AUS = {
					set_country_flag = GER_appealed_old_friends
					country_event = { id = gerfor.51 days = 1 }
				}
			}
			else_if = {
				limit = { FROM = { tag = BUL } }
				BUL = {
					set_country_flag = GER_appealed_old_friends
					country_event = { id = gerfor.53 days = 1 }
				}
			}
		}
	}

french_occupation_decisions = {

	GER_erode_cof_legitimacy = {
		icon = eng_propaganda_campaigns

		allowed = {
			NOT = { tag = FRA }
		}

		available = {
			FRA = {
				is_government_in_exile = yes
				has_legitimacy > 20
			}
		}

		visible = {
			has_global_flag = FRA_capitulated
			FRA = { is_government_in_exile = yes }
			has_war_with = FRA
			is_subject = no

			set_temp_variable = { cof_states = 0 }
			FRA = {
				all_owned_state = {
					if = {
						limit = {
							is_core_of = FRA
							is_controlled_by = ROOT
						}
						add_to_temp_variable = { var = ROOT.cof_states value = 1 }
					}
				}
			}
			check_variable = { cof_states > 2 }
		}

		fire_only_once = no
		days_remove = 30
		cost = 25

		ai_will_do = { factor = 200 }

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete GER_erode_cof_legitimacy"
			set_country_flag = GER_eroding_cof_leg
		}

		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove GER_erode_cof_legitimacy"
			clr_country_flag = GER_eroding_cof_leg
			if = {
				limit = { tag = NFA }
				FRA = {
					add_legitimacy = -25
					add_war_support = -0.15
				}
			}
			else_if = {
				limit = {
					country_exists = NFA
					NOT = { tag = GER }
				}
				FRA = {
					add_legitimacy = -15
					add_war_support = -0.10
				}
			}
			else = {
				FRA = {
					add_legitimacy = -10
					add_war_support = -0.05
				}
			}
		}
	}

	GER_invite_nfa_to_france = {
		icon = generic_agreement

		allowed = { tag = GER }

		available = {
			NOT = { has_war_with = NFA }
		}

		visible = {
			has_war_with = FRA
			has_global_flag = FRA_capitulated
			FRA = { is_government_in_exile = yes }
			FROM = {
				has_war_with = FRA
				NOT = { has_war_with = ROOT }
			}

			set_temp_variable = { cof_states = 0 }
			set_temp_variable = { nfa_states = 0 }
			FRA = {
				all_owned_state = {
					if = {
						limit = {
							is_core_of = FRA
							is_controlled_by = GER
							region = 20
						}
						add_to_temp_variable = { ROOT.cof_states = 1 }
					}
					if = {
						limit = {
							is_core_of = FRA
							is_controlled_by = NFA
						}
						add_to_temp_variable = { ROOT.nfa_states = 1 }
						subtract_from_temp_variable = { ROOT.cof_states = 1 }
					}
				}
			}
			check_variable = { cof_states > 0 }
			check_variable = { nfa_states < 3 }
		}

		target_root_trigger = {
			has_war_with = FRA
		}

		targets = { NFA }

		target_trigger = {
			FROM = {
				has_war_with = FRA
				NOT = { has_war_with = ROOT }
			}
		}

		fire_only_once = yes
		cost = 0

		ai_will_do = {
			factor = 10
			modifier = {
				factor = 0
				has_country_flag = GER_eroding_cof_leg
			}
			modifier = {
				factor = 0
				has_political_power > 50
			}
			modifier = {
				factor = 0
				FRA = { has_legitimacy < 50 }
			}
			modifier = {
				factor = 0
				OR = {
					NOT = { country_exists = ENG }
					ENG = { surrender_progress > 0.2 }
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete GER_invite_nfa_to_france target: [From.GetName]"
			NFA = { country_event = capt.43 }
		}
	}
	UKR_GER_invade_ukraine = { #in case of syndicalist Ukraine
		icon = RUS_Attack_Region

		allowed = { always = no }
		visible = {
			UKR = { exists = yes }
			has_country_flag = UKR_invasion
			NOT = { has_war_with = UKR }
			can_declare_war_on = FROM
		}
		cancel_trigger = {
			has_war_with = UKR
		}

		targets = { UKR RUS }
		target_root_trigger = {
			has_country_flag = UKR_invasion
		}
		target_trigger = {
			can_declare_war_on = FROM
		}

		days_remove = 45
		war_with_target_on_remove = yes
		fire_only_once = yes

		ai_will_do = { factor = 1000 }

		cancel_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision cancel UKR_GER_invade_ukraine target: [From.GetName]"
			clear_decision_attack_AI = yes
			clr_country_flag = UKR_invasion
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision complete UKR_GER_invade_ukraine target: [From.GetName]"
			setup_decision_attack_AI = yes
			country_event = { id = ukraine_borotbist.101 days = 3 }
			hidden_effect = {
				INT = { country_event = { id = ukraine_borotbist.105 days = 60 } }
				UKR = { country_event = { id = ukraine_borotbist.106 days = 90 } }
			}
		}
		remove_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision remove UKR_GER_invade_ukraine target: [From.GetName]"
			clear_decision_attack_AI = yes
			clr_country_flag = UKR_invasion
		}
	}
