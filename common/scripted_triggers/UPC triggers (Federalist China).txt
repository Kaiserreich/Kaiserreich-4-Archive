############################################
#### Federalist China Scripted Triggers ####
############################################

UPC_is_potential_UPC_tag_trigger = {
	OR = {
		tag = GXC
		tag = HNN
		tag = SZC
		tag = YUN
	}
}

UPC_incumbent_is_GRA_trigger = {
	custom_trigger_tooltip = {
		tooltip = UPC_incumbent_is_GRA_trigger_tt
		OR = {
			#GXC Leaders
			has_country_leader = { character = GXC_chen_jiongming }
			has_country_leader = { character = GXC_chen_qiyou }

			#HNN Leaders
			has_country_leader = { character = HNN_zhao_hengti }

			#YUN Leaders
			has_country_leader = { character = YUN_tang_jiyao }
			has_country_leader = { character = YUN_tang_xiaoying }

			#SZC Leaders
			has_country_leader = { character = SZC_dan_maoxin }
			has_country_leader = { character = SZC_liu_wenhui }
		}
	}
}

UPC_incumbent_is_CDL_trigger = {
	custom_trigger_tooltip = {
		tooltip = UPC_incumbent_is_CDL_trigger_tt
		OR = {
			#GXC Leaders
			has_country_leader = { character = GXC_zhang_junmai }

			#YUN Leaders
			has_country_leader = { character = YUN_luo_longji }
			has_country_leader = { character = YUN_cun_shusheng }

			#SZC Leaders
			has_country_leader = { character = SZC_zhang_lan }
		}
	}
}

UPC_has_governing_majority_trigger = {
	custom_trigger_tooltip = {
		tooltip = UPC_governing_situation_majority_key
		OR = {
			AND = {
				UPC_incumbent_is_GRA_trigger = yes
				check_variable = { global.UPC_GRA_needed_votes_for_majority_var < 1 }
			}
			AND = {
				UPC_incumbent_is_CDL_trigger = yes
				check_variable = { global.UPC_CDL_needed_votes_for_majority_var < 1 }
			}
		}
	}
}
