is_controlled_by_BP = {
	controller = {
		OR = {
			is_in_faction_with = SER
			tag = SER
			tag = ROM
			tag = BUL
			tag = ITA
			tag = VNC
			tag = POL
			tag = CZE
			tag = GAL
			tag = CRO
		}
		if = {
			limit = { is_in_faction_with = SER }
			SER = { is_valid_peace_target = yes }
		}
		else_if = {
			limit = { is_ally_with = BUL }
			BUL = { is_valid_peace_target = yes }
		}
		else = {
			is_valid_peace_target = yes
		}
	}
}
