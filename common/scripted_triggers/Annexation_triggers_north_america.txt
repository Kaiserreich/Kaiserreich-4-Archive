#Alaska - annexation_ALASKA_CONQ
can_release_alaska = {
	state = 463 #Alaska
	NOT = { is_claimed_by = ROOT }
	ROOT = {
		should_release_america_first = yes
		# Do Canadian release first
		OR = {
			country_exists = CAN
			has_country_flag = CAN_CONQ
			739 = { is_valid_annexation_state = no } #Ottawa
		}
		OR = {
			var:361.owner = { #Maryland
				is_american_tag = yes
				ROOT_can_grant_land = yes
			}
			is_ally_with = CAN
		}
		NOT = {
			tag = CAN
			tag = QUE
			is_american_tag = yes
			has_country_flag = ALASKA_CONQ
		}
	}
}

#United States - annexation_USA_CONQ
can_release_america = {
	if = {
		limit = {
			ROOT = { tag = CAN }
		}
		ROOT = {
			any_owned_state = {
				is_mainland_america = yes
				NOT = { is_claimed_by = ROOT }
				NOT = {
					AND = {
						ROOT = { has_country_flag = CAN_seized_new_england }
						OR = {
							region = 186 #New England
							state = 796 #Upstate New York
						}
					}
					AND = {
						ROOT = { has_country_flag = CAN_seized_alaska }
						state = 463 #Alaska
					}
				}
			}
		}
	}
	else = {
		OR = {
			state = 361 #Maryland
			state = 371 #Louisiana
			state = 395 #Illinois
		}
		371 = { is_valid_annexation_state = yes }
		395 = { is_valid_annexation_state = yes }
		361 = { is_valid_annexation_state = yes }
	}
	NOT = { is_claimed_by = ROOT }
	ROOT = {
		is_american_tag = no
		NOT = {
			if = {
				limit = {
					NOT = { tag = CAN }
				}
				country_exists = USA
				country_exists = CSA
				country_exists = TEX
				country_exists = CAL
			}
			has_country_flag = USA_CONQ
			any_enemy_country = { is_american_tag = yes }
		}
	}
}

#Danish North Atlantic - annexation_GRN_CONQ
can_release_greenland = {
	state = 101 #Greenland
	NOT = { is_claimed_by = ROOT }
	ROOT = {
		# Do Danish release first
		OR = {
			country_exists = DEN
			has_country_flag = DEN_CONQ
			37 = { is_valid_annexation_state = no } #Zealand
		}
		# Do Norwegian release first
		OR = {
			country_exists = NOR
			has_country_flag = NOR_CONQ
			110 = { is_valid_annexation_state = no } #Ostlandet
		}
		# Do Canadian release first
		OR = {
			country_exists = CAN
			has_country_flag = CAN_CONQ
			tag = QUE
			739 = { is_valid_annexation_state = no } #Ottawa
		}
		NOT = {
			tag = SCA
			tag = DEN
			tag = NOR
			has_country_flag = GRN_CONQ
			has_country_flag = swe_flag_sweden_norway
			AND = {
				tag = CAN
				is_faction_leader = yes
			}
		}
	}
	OR = {
		DEN = { is_ally_of_ROOT = yes }
		NOR = { is_ally_of_ROOT = yes }
		SCA = { is_ally_of_ROOT = yes }
		SWE = {
			has_country_flag = swe_flag_sweden_norway
			is_ally_of_ROOT = yes
		}
		CAN = {
			is_ally_of_ROOT = yes
			NOT = { owns_state = 101 }
			OR = {
				is_faction_leader = yes
				is_subject = yes
				NOT = {
					DEN = { is_ally_of_ROOT = yes }
					SCA = { is_ally_of_ROOT = yes }
				}
			}
		}
	}
}
