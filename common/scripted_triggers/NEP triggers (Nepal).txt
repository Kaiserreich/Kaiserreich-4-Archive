NEP_conspiracy_ai_trigger = {
	NEP = { is_ai = yes }
	HND = { is_ai = yes }
	RAJ = { is_ai = yes }
}

NEP_HND_influence_winning = {
	NOT = { check_variable = { NEP.NEP_relative_HND_influence < 25 } }
}

NEP_HND_influence_winning_extra = {
	NOT = { check_variable = { NEP.NEP_relative_HND_influence < 30 } }
}

NEP_HND_influence_higher = {
	check_variable = { NEP.NEP_relative_HND_influence > 0 }
}

NEP_RAJ_influence_winning = {
	NOT = { check_variable = { NEP.NEP_relative_RAJ_influence < 25 } }
}

NEP_RAJ_influence_winning_extra = {
	NOT = { check_variable = { NEP.NEP_relative_RAJ_influence < 30 } }
}

NEP_RAJ_influence_higher = {
	check_variable = { NEP.NEP_relative_RAJ_influence > 0 }
}

		has_country_flag = NEP_conspiracy_HND_ai
		AND = {
			has_country_flag = NEP_conspiracy_player_HND_ai
			NEP_HND_influence_winning_extra = no
		}
#		AND = {
#			has_country_flag = NEP_conspiracy_player_RAJ_ai
#			NEP_RAJ_influence_winning_extra = yes
#		}
		AND = {
			has_country_flag = NEP_conspiracy_NEP_ai
			NEP_HND_influence_winning = yes
		}
	}
}

NEP_increase_RAJ_trigger = {

NEP_british_india_is_valid_relation_target = {
	country_exists = RAJ
	RAJ = {
		is_subject = no
		has_capitulated = no
		NOT = { has_war_with = ROOT }
	}
}

NEP_republican_india_is_valid_relation_target = {
	if = {
		limit = {
			NOT = { country_exists = HND }
			country_exists = PRF
		}
		country_exists = PRF #repeated for the sake of player clarity
		PRF = {
			is_subject = no
			has_capitulated = no
			NOT = { has_war_with = ROOT }
		}
	}
	else = {
		country_exists = HND
		HND = {
			is_subject = no
			has_capitulated = no
			NOT = { has_war_with = ROOT }
		}
	}
}
