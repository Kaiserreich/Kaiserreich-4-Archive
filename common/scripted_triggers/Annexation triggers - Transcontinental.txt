#West Indies Federation - both Central and South America - annexation_WIF_CONQ
can_release_west_indies = {
	OR = {
		state = 309 #Paramaribo
		state = 686 #Puerto Rico
		state = 687 #West Guyana
		state = 689 #Jamaica
		state = 691 #Trinidad & Tobago
	}
	count_triggers = {
		amount = 3
		309 = { is_valid_annexation_state = yes } #Paramaribo
		686 = { is_valid_annexation_state = yes } #Puerto Rico
		687 = { is_valid_annexation_state = yes } #Guyana
		689 = { is_valid_annexation_state = yes } #Jamaica
		691 = { is_valid_annexation_state = yes } #Trinidad
	}
	ROOT = {
		NOT = {
			has_country_flag = WIF_CONQ
			country_exists = WIF
		}
		if = {
			limit = { tag = GBR }
			has_country_flag = GBR_decolonize
		}
	}
}

#Aegean Coast - annexation_AEGEAN_CONQ
can_release_aegean_coast = {
	OR = {
		state = 164 #Dodecanese
		state = 187 #Aegean Islands
		state = 339 #Smyrna
		state = 340 #Bursa
		state = 342 #Mugla
	}
	NOT = { is_claimed_by = ROOT }
	ROOT = {
		should_release_turkey_first = yes
		should_release_greece_first = yes
		OR = {
			tag = TUR
			tag = GRE
			TUR = { ROOT_can_grant_land = yes }
			GRE = { ROOT_can_grant_land = yes }
		}
		NOT = { has_country_flag = AEGEAN_CONQ }
	}
}

				689 = { is_valid_annexation_state = yes } #Jamaica
				691 = { is_valid_annexation_state = yes } #Trinidad
			}

	}
}
