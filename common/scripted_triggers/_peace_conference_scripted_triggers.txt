is_important_state = {
	OR = {
		state = 118 #Gibraltar
		state = 753 #Suez Canal
		state = 685 #Panama Canal
		state = 116 #Malta
		state = 336 #Singapore
	}
}

is_ambition_area_of_ROOT = {
	OR = {
		has_variable = ambition_area_@ROOT
		AND = {
			ROOT = {
				is_actual_major = yes
				OR = {
					tag = GER
					tag = CAN
					tag = GBR
					tag = NFA
					tag = FRA
					tag = JAP
				}
			}
			is_important_state = yes
		}
	}
}

is_ambition_area_of_PREV = {
	OR = {
		has_variable = ambition_area_@PREV
		AND = {
			PREV = {
				is_actual_major = yes
				OR = {
					tag = GER
					tag = CAN
					tag = GBR
					tag = NFA
					tag = FRA
					tag = JAP
				}
			}
			is_important_state = yes
		}
	}
}

is_claim_or_core_or_ambition_of_ROOT = {

		is_claimed_by = ROOT
