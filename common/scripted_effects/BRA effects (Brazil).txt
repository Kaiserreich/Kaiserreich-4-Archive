BRA_change_soft = {
	if = {
		limit = { BRA_army_soft = yes }
		# convert percent to absolute
		if = {
			limit = {
				check_variable = { BRA_soft_change > -1 }
				check_variable = { BRA_soft_change < 1 }
			}
			multiply_temp_variable = { BRA_soft_change = 100 }
		}

		# tooltip
		effect_tooltip = {
			add_to_variable = {
				BRA_reform_soft = BRA_soft_change
				tooltip = BRA_soft_tooltip
			}
		}

		# lower limit = legitimacy * -1
		set_temp_variable = { BRA_lower_limit = BRA_reform_soft }
		multiply_temp_variable = { BRA_lower_limit = -1 }

		# upper limit = 100 - legitimacy
		set_temp_variable = { BRA_upper_limit = 100 }
		subtract_from_temp_variable = { BRA_upper_limit = BRA_reform_soft }

		# avoid values outside the allowed range
		clamp_temp_variable = {
			var = BRA_soft_change
			min = BRA_lower_limit
			max = BRA_upper_limit
		}

		# add the variable
		add_to_variable = { BRA_reform_soft = BRA_soft_change }
	}
}
