YUN_tang_turns_federalist = {
	set_country_flag = YUN_federalists_chosen
	YUN_tang_jiyao = {
		if = {
			limit = { has_ideology = authoritarian_democrat_subtype }
			promote_character = authoritarian_democrat_subtype
		}
		else = {
			add_country_leader_role = {
				promote_leader = yes
				country_leader = {
					desc = YUN_tang_jiyao_leader_desc
					ideology = authoritarian_democrat_subtype
				}
			}
		}
	}
	set_politics = {
		ruling_party = authoritarian_democrat
		elections_allowed = no
		name = YUN_Tang_Federalist_party
		long_name = YUN_Tang_Federalist_party_long
	}
	add_popularity = {
		ideology = paternal_autocrat
		popularity = -0.4
	}
	add_popularity = {
		ideology = authoritarian_democrat
		popularity = 0.5
	}
}

YUN_add_integration = {
	hidden_effect = {
		if = {
			limit = { NOT = { has_dynamic_modifier = { modifier = YUN_integration_dynamic_modifier } } }
			add_dynamic_modifier = { modifier = YUN_integration_dynamic_modifier }
		}
	}
}

YUN_remove_integration = {
	hidden_effect = {
		clear_variable = YUN_integration_var
		if = {
			limit = { has_dynamic_modifier = { modifier = YUN_integration_dynamic_modifier } }
			remove_dynamic_modifier = { modifier = YUN_integration_dynamic_modifier }
		}
	}
}

RKMT_set_leadership_effect = {
	if = {
		limit = {
			OR = {
				GXC = {
					exists = yes
					is_subject = no
					has_civil_war = no
					has_capitulated = no
					NOT = { any_enemy_country = { is_ally_with = YUN } }
					china_is_aligned_with_right_kmt = yes
				}
				AND = {
					GXC = {
						exists = yes
						is_subject = no
						has_civil_war = no
						has_capitulated = no
						NOT = { any_enemy_country = { is_ally_with = YUN } }
						china_is_aligned_with_left_kmt = yes
					}
					CHI = { exists = no }
				}
			}
			YUN = {
				exists = yes
				is_subject = no
				has_civil_war = no
				has_capitulated = no
				NOT = { any_enemy_country = { is_ally_with = GXC } }
				china_is_aligned_with_kmt = yes
			}
		}
		if = {
			limit = {
				GXC = { is_ai = yes }
				YUN = { china_is_aligned_with_right_kmt = yes }
				YUN = { is_ai = no }
			}
			YUN = {
				set_temp_variable = { target = GXC }
				china_puppet_TARGET_effect = yes #YUN will puppet GXC if YUN is a player and RKMT
				set_potential_national_government = yes
			}
		}
		else_if = {
			limit = {
				GXC = { is_ai = yes }
				GXC = { china_is_aligned_with_left_kmt = yes }
				GXC = { china_is_potential_government = no }
				YUN = { china_is_aligned_with_right_kmt = yes }
			}
			YUN = {
				set_temp_variable = { target = GXC }
				china_puppet_TARGET_effect = yes #YUN will puppet GXC if YUN is RKMT and GXC is ai and LKMT and hasn't formed a government yet since YUN formed a government first in this scenario
				set_potential_national_government = yes
			}
		}
		else = {
			GXC = {
				set_temp_variable = { target = YUN }
				china_puppet_TARGET_effect = yes #GXC will puppet YUN if YUN is ai and GXC is RKMT
				set_potential_national_government = yes
			}
		}
	}
}
