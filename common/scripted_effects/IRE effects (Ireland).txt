IRE_nationalist_discontent_increase = {
	if = {
		limit = { has_variable = IRE_nationalist_discontent }
		set_temp_variable = { IRE_nationalist_discontent_delta_tt = IRE_nationalist_discontent_delta }
		multiply_temp_variable = { IRE_nationalist_discontent_delta_tt = 100 }
		custom_effect_tooltip = IRE_nationalist_discontent_increase_tt
		subtract_from_variable = { IRE_nationalist_discontent = IRE_nationalist_discontent_delta }
	}
}

IRE_remove_ulster_privileges = {
	if = {
		limit = {
			check_variable = { IRE_nationalist_discontent = 0 }
			check_variable = { IRE_loyalist_anger = 0 }
			if = {
				limit = { has_country_flag = IRE_NIR_revolt_guaranteed }
				has_global_flag = IRE_had_NIR_revolt
			}
		}
		hidden_effect = {
			country_event = ireland.2608
			remove_dynamic_modifier = { modifier = IRE_ulster_privileges_modifier }
			clear_variable = IRE_nationalist_discontent
			clear_variable = IRE_loyalist_anger
			clear_variable = IRE_loyalist_fear
			clear_variable = IRE_improve_nationalist_discontent_cost
			clear_variable = IRE_improve_loyalist_anger_cost
			clear_variable = IRE_decision_duration
			if = {
				limit = { has_completed_focus = IRE_draft_new_constitution }
				remove_targeted_decision = { target = 119 decision = IRE_improve_nationalist_discontent }
				remove_targeted_decision = { target = 119 decision = IRE_improve_loyalist_anger }
			}
			if = {
				limit = { has_country_flag = IRE_nir_resistance_started }
				119 = {
					force_enable_resistance = { occupier = IRE clear = yes }
					remove_dynamic_modifier = { modifier = IRE_ulster_resistance_modifier }
					remove_dynamic_modifier = { modifier = IRE_soc_ulster_resistance_modifier }
					set_demilitarized_zone = no
				}
				if = {
					limit = {
						OR = {
							IRE_has_puppet_govt = yes
							NOT = { owns_state = 119 }
						}
					}
					clr_country_flag = IRE_nir_resistance_started
				}
				remove_ideas = IRE_ulster_privileges_idea
				clr_country_flag = IRE_has_improved_anger
				clr_country_flag = IRE_has_improved_discontent
				clr_country_flag = IRE_NIR_revolt_available
				clr_country_flag = IRE_NIR_revolt_guaranteed
				clr_country_flag = IRE_NIR_revolt_avoided
				clear_variable = IRE_NIR_decisions_cost
				clear_variable = IRE_NIR_decisions_time
				clear_variable = IRE_NIR_mission_time
				clear_variable = IRE_NIR_decisions_time_long
				clear_variable = IRE_NIR_decisions_variable
				clear_variable = IRE_NIR_mission_time_long
				clear_variable = IRE_NIR_mission_remove
				clear_variable = IRE_ulster_resistance_delta
				clear_variable = IRE_ulster_resistance
				remove_mission = IRE_NIR_mission
				remove_mission = IRE_soc_NIR_mission
				remove_mission = IRE_improve_discontent_mission
				remove_mission = IRE_improve_anger_mission
			}
		}
	}
}

IRE_bop_result = {
	if = {
		limit = {
			OR = {
				AND = {
					is_ai = no
					power_balance_value = {
						id = IRE_an_power_balance
						value < -0.24 # Negative for EOD, positive for GOC
					}
				}
				AND = {
					is_ai = yes
					has_country_flag = IRE_polpath_parliamentary_an
					NOT = { has_game_rule = { rule = IRE_path option = IRE_PATH_SOCCON } }
					NOT = { has_game_rule = { rule = IRE_path option = IRE_PATH_AUTDEM } }
				}
			}
		}
		set_country_flag = IRE_bop_eod
	}
	else_if = {
		limit = {
			OR = {
				AND = {
					is_ai = no
					power_balance_value = {
						id = IRE_an_power_balance
						value > 0.24 # Negative for EOD, positive for GOC
					}
				}
				AND = {
					is_ai = yes
					has_country_flag = IRE_polpath_abstentionist_an
					NOT = { has_game_rule = { rule = IRE_path option = IRE_PATH_SOCCON } }
					NOT = { has_game_rule = { rule = IRE_path option = IRE_PATH_AUTDEM } }
				}
			}
		}
		set_country_flag = IRE_bop_goc
	}
	else = {
		set_country_flag = IRE_bop_jjok
	}
}

IRE_ruling_government_popularity_change = {
	if = {
		limit = { IRE_has_one_party_govt = no }
		set_temp_variable = { IRE_main_party_change = ideology_change_var }
		multiply_temp_variable = { IRE_main_party_change = 0.7 }
		set_temp_variable = { IRE_smaller_party_change = ideology_change_var }
		multiply_temp_variable = { IRE_smaller_party_change = 0.4 }
		if = {
			limit = { IRE_has_three_party_govt = yes }
			multiply_temp_variable = { IRE_smaller_party_change = 0.5 }
		}
		set_temp_variable = { ideology_change_var = IRE_main_party_change }
		change_ruling_party_popularity = yes
		if = {
			limit = {
				IRE_has_soccon_in_coalition = yes
				NOT = { has_government = social_conservative }
			}
			add_popularity = { ideology = social_conservative popularity = IRE_smaller_party_change }
		}
		if = {
			limit = {
				IRE_has_socdem_in_coalition = yes
				NOT = { has_government = social_democrat }
			}
			add_popularity = { ideology = social_democrat popularity = IRE_smaller_party_change }
		}
		if = {
			limit = {
				IRE_has_marlib_in_coalition = yes
				NOT = { has_government = market_liberal }
			}
			add_popularity = { ideology = market_liberal popularity = IRE_smaller_party_change }
		}
		if = {
			limit = {
				IRE_has_soclib_in_coalition = yes
				NOT = { has_government = social_liberal }
			}
			add_popularity = { ideology = social_liberal popularity = IRE_smaller_party_change }
		}
		if = {
			limit = {
				IRE_has_pataut_in_coalition = yes
				has_authoritarian_government = no
			}
			if = {
				limit = { NOT = { has_country_flag = IRE_an_autdem } }
				add_popularity = { ideology = paternal_autocrat popularity = IRE_smaller_party_change }
			}
			else = {
				add_popularity = { ideology = authoritarian_democrat popularity = IRE_smaller_party_change }
			}
		}
		if = {
			limit = { has_country_flag = IRE_govt_blueshirt_coup }
			add_popularity = { ideology = national_populist popularity = IRE_smaller_party_change }
		}
	}

		change_ruling_party_popularity = yes
	}
}
