# Cedes land and grants cores for the Belgrade Pact members
belgrade_pact_gain_land_from_bulgaria = {
	if = {
		limit = { 767 = { is_claimed_by = BUL } } #Nis
		767 = { remove_claim_by = BUL }
	}

	# give Ottoman lands if they're occupied
	if = {
		limit = {
			OTT = { has_country_flag = OTT_war_for_thrace }
		}
		every_state = {
			limit = {
				is_owned_by = BUL
				has_state_flag = SER_transfer_to_OTT
			}
			transfer_state_to = OTT
			if = {
				limit = { 341 = { OTT_is_autonomous_region = yes } } #Adrianople
				ott_set_authority_vilayet_low = yes
			}
			else_if = {
				limit = { 341 = { OTT_is_vilayet = yes } } #Adrianople
				ott_set_authority_vilayet_medium = yes
			}
			else = {
				ott_set_authority_vilayet_high = yes
			}
		}
	}

	# give Serbian lands
	every_state = {
		limit = {
			is_owned_by = BUL
			OR = {
				is_core_of = SER
				is_claimed_by = SER
				state = 106 #Macedonia
			}
		}
		if = {
			limit = {
				NOT = { is_claimed_by = SER }
				NOT = { is_core_of = SER }
			}
			add_claim_by = SER
		}
		transfer_state_to = SER
		remove_core_of = BUL
		add_claim_by = BUL
	}
	hidden_effect = {
		BUL = {
			country_event = { id = bulgaria.29 days = 3 }
			country_event = { id = serbia.22 days = 7 random_days = 3 }
		}
		SER = {
			country_event = { id = serbia.23 days = 14 random_days = 7 }
		}
	}

	if = {
		limit = {
			106 = { is_owned_by = SER }
			731 = { is_owned_by = BUL }
			GRE = {
				NOT = {
					is_in_faction_with = SER
					has_war_with = BUL
				}
			}
		}
		SER = { transfer_state = 731 }
		731 = {
			remove_core_of = BUL
			add_claim_by = BUL
		}
	}

	# give Romanian lands
	if = {
		limit = {
			ROM = {
				is_in_faction_with = SER
			}
		}
		77 = {
			if = {
				limit = { is_owned_by = BUL }
				transfer_state_to = ROM
			}
			if = {
				limit = {
					NOT = { is_claimed_by = ROM }
					NOT = { is_core_of = ROM }
				}
				add_claim_by = ROM
			}
			add_compliance = 25
		}
		every_state = {
			limit = {
				is_owned_by = BUL
				NOT = { state = 77 } #Southern Dobruja
				OR = {
					is_core_of = ROM
					is_claimed_by = ROM
					state = 772 #Northern Dobruja
				}
			}
			if = {
				limit = {
					NOT = { is_claimed_by = ROM }
					NOT = { is_core_of = ROM }
				}
				add_core_of = ROM
			}
			if = {
				limit = { is_claimed_by = ROM }
				hidden_effect = { remove_claim_by = ROM }
				add_core_of = ROM
			}
			transfer_state_to = ROM
			remove_core_of = BUL
			add_claim_by = BUL
		}
	}

	# give Greek lands
	if = {
		limit = { GRE = { is_in_faction_with = SER } }
		every_state = {
			limit = {
				is_owned_by = BUL
				NOT = { has_state_flag = SER_transfer_to_OTT }
				OR = {
					is_core_of = GRE
					state = 731 #Western Salonica
					state = 769 #Eastern Salonica
					state = 184 #Western Thrace
				}
			}
			if = {
				limit = {
					NOT = { is_claimed_by = GRE }
					NOT = { is_core_of = GRE }
				}
				add_claim_by = GRE
			}
			transfer_state_to = GRE
			remove_core_of = BUL
			add_claim_by = BUL
		}
	}
}

belgrade_pact_occupies_bulgaria = {
	# Serbia gains Sofia and Pleven, and their cores
	every_owned_state = {
		limit = {
			is_core_of = BUL
			is_not_claim_or_core_of_ROOT = yes
			OR = {
				state = 48 # Sofia
				state = 770 # Pleven
				state = 990 # Vidin
				is_core_of = SER
				is_claimed_by = SER
			}
		}
		transfer_state_to = SER
	}

	# Greece gains Plovdiv and their cores
	if = {
		limit = {
			NOT = { tag = GRE }
			GRE = { is_in_faction_with = SER }
		}
		every_owned_state = {
			limit = {
				is_core_of = BUL
				is_not_claim_or_core_of_ROOT = yes
				OR = {
					state = 212 # Plovdiv
					state = 1089 # Momchilgrad
					is_core_of = GRE
					is_claimed_by = GRE
				}
			}
			transfer_state_to = GRE
		}
	}

	# Romania gains Varna and Burgas, and their cores
	if = {
		limit = {
			NOT = { tag = ROM }
			ROM = { is_in_faction_with = SER }
		}
		every_owned_state = {
			limit = {
				is_core_of = BUL
				is_not_claim_or_core_of_ROOT = yes
				OR = {
					state = 771 # Varna
					state = 211 # Burgas
					is_core_of = ROM
					is_claimed_by = ROM
				}
			}
			transfer_state_to = ROM
		}
	}
}

belgrade_pact_loot_factory_from_bulgaria = {
	if = {
		limit = { NOT = { has_global_flag = belgrade_pact_took_bulgarian_factories } }
		set_global_flag = belgrade_pact_took_bulgarian_factories
		# Bulgaria loses one civilian and one military factory
		BUL = {
			if = {
				limit = {
					any_owned_state = {
						is_controlled_by = BUL
						industrial_complex > 0
					}
				}
				ROOT = { set_country_flag = gets_ic }
				random_owned_controlled_state = {
					limit = { industrial_complex > 0 }
					remove_building = { type = industrial_complex level = 1 }
				}
			}
			if = {
				limit = {
					any_owned_state = {
						is_controlled_by = BUL
						arms_factory > 0
					}
				}
				ROOT = { set_country_flag = gets_af }
				random_owned_controlled_state = {
					limit = { arms_factory > 0 }
					remove_building = { type = arms_factory level = 1 }
				}
			}
		}

		# ROOT gains two factories
		if = {
			limit = {
				OR = {
					BUL = {
						any_owned_state = {
							is_controlled_by = BUL
							industrial_complex > 0
						}
					}
					has_country_flag = gets_ic
				}
			}
			clr_country_flag = gets_ic
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					industrial_complex > 0
					free_building_slots = {
						building = industrial_complex
						size > 0
						include_locked = yes
					}
				}
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = industrial_complex
					level = 1
					instant_build = yes
				}
			}
		}
		if = {
			limit = {
				OR = {
					BUL = {
						any_owned_state = {
							is_controlled_by = BUL
							arms_factory > 0
						}
					}
					has_country_flag = gets_af
				}
			}
			clr_country_flag = gets_af
			random_owned_controlled_state = {
				limit = {
					is_core_of = ROOT
					arms_factory > 0
					free_building_slots = {
						building = arms_factory
						size > 0
						include_locked = yes
					}
				}
				add_extra_state_shared_building_slots = 1
				add_building_construction = {
					type = arms_factory
					level = 1
					instant_build = yes
				}
			}
		}
	}
}

belgrade_pact_gain_land_from_albania = {
	if = {
		limit = {
			766 = { is_owned_by = ALB } #Northern Epirus
			GRE = { is_in_faction_with = SER }
		}
		GRE = {
			transfer_state = 766 # Northern Epirus
			hidden_effect = {
				add_state_core = 766 # Northern Epirus
			}
		}
	}
}
