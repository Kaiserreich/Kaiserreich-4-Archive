CAM_create_sisowath_leaders = {
	CAM_set_monarchy_cosmetic_tag = yes
	if = { #Sisowath Monivong
		limit = {
			date < 1941.4.24
			CAM = { CAM_sisowath_monivong = { is_dead = no } }
		}
		CAM = {
			CAM_sisowath_monivong = {
				character_becomes_leader_of_ruling_party_effect_tooltip = yes
				hidden_effect = {
					set_nationality = PREV.PREV
					remove_all_country_leader_roles = yes
					add_country_leader_role = {
						country_leader = { ideology = social_democrat_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = social_liberal_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = market_liberal_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = social_conservative_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = authoritarian_democrat_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = paternal_autocrat_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = national_populist_subtype }
						promote_leader = yes
					}
				}
			}
		}
	}
	else = { #Sisowath Monireth
		CAM = {
			hidden_effect = { CAM_sisowath_monivong = { kill_character = yes } }
			CAM_sisowath_monireth = {
				character_becomes_leader_of_ruling_party_effect_tooltip = yes
				hidden_effect = {
					set_nationality = PREV.PREV

						country_leader = { ideology = social_democrat_subtype }
						promote_leader = yes

						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = market_liberal_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = social_conservative_subtype }
						promote_leader = yes
					}
					add_country_leader_role = {
						country_leader = { ideology = authoritarian_democrat_subtype }
						promote_leader = yes

					add_country_leader_role = {

CAM_set_monarchy_cosmetic_tag = {
	if = {
		limit = { NOT = { has_cosmetic_tag = CAM_monarchy } }
		set_cosmetic_tag = CAM_monarchy
	}
}

CAM_remove_monarchy = {
	if = {
		limit = { has_cosmetic_tag = CAM_monarchy }
		drop_cosmetic_tag = yes
	}
	if = {
		limit = { has_character = CAM_sisowath_monivong }
		CAM_sisowath_monivong = {
			hidden_effect = { remove_all_country_leader_roles = yes }
			retire = yes
		}
	}
	if = {
		limit = { has_character = CAM_sisowath_monireth }
		CAM_sisowath_monireth = {
			hidden_effect = { remove_all_country_leader_roles = yes }
			retire = yes
		}
	}
	if = {
		limit = { has_character = CAM_norodom_suramarit }

			hidden_effect = { remove_all_country_leader_roles = yes }

		}
	}
