	on_monthly_UKR = {
		effect = {
			# UKR has 60%+ surrender progress
			if = {
				limit = {
					has_war_with = RUS

	on_peaceconference_started = { #Access to postwar content
		effect = {
			if = {

					ROOT = { tag = UKR }

				}

					if = {
						limit = {
							has_government = paternal_autocrat

					}
					else_if = {
						limit = {
							has_government = authoritarian_democrat

	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				limit = { FROM.FROM = { has_state_flag = UKR_claim_state } }
				if = {
					limit = {
						is_ally_with = UKR
						UKR = { FROM.FROM = { is_claim_or_core_of_PREV = no } }
					}
					UKR = { activate_targeted_decision = { decision = UKR_claim_state target = FROM.FROM } }
				}
				else = {
					UKR = { remove_targeted_decision = { decision = UKR_claim_state target = FROM.FROM } }
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
