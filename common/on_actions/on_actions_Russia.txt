on_actions = {
	on_startup = {
		effect = {
			#Russian integration arrays
			for_each_scope_loop = {
				array = RUS.owned_states
				add_to_array = { RUS.RUS_starting_states_array = THIS }
				if = {
					limit = {
						is_siberia = yes
						NOT = { is_core_of = TRM }
					}
					add_to_array = { TRM.TRM_integration_array = THIS }
				}
			}
			for_each_scope_loop = {
				array = DKB.owned_states
				add_to_array = { RUS.RUS_starting_states_array = THIS }
			}
			add_to_array = { RUS.RUS_starting_states_array = 408.id } #Vladivostok
			add_to_array = { RUS.RUS_starting_states_array = 409.id } #Khabarovsk
			880 = { #Dvinsk
				add_dynamic_modifier = { modifier = BAT_Latgale_autonomous_region }
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
				limit = {
					FROM = { tag = TRM }
				}
				every_state = {
					limit = { is_claim_or_core_of_FROM = yes }
					remove_claim_by = FROM
					remove_core_of = FROM
				}
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation = {
		effect = {
			# Disable SOV civil war unit lock
			if = {
				limit = {
					ROOT = {
						tag = SOV
						has_idea = SOV_chaotic_revolution
					}
				}
				ROOT = {
					remove_ideas = SOV_chaotic_revolution
					country_lock_all_division_template = no
				}
			}
			# Disband the Russian Academy of Sciences
			if = {
				limit = {
					tag = RUS
					is_in_tech_sharing_group = russian_research
				}
				remove_from_tech_sharing_group = russian_research
				every_other_country = {
					limit = { is_in_tech_sharing_group = russian_research }
					remove_from_tech_sharing_group = russian_research
				}
			}
		}
	}

	#ROOT is winner #FROM gets annexed - This fires just before FROM gets annexed, meaning the country and everything it owns still exists. It will also fire on_annex and on_civil_war_end
	on_civil_war_end_before_annexation = {
		effect = {
			if = {
				limit = { is_russian_tag = yes }
				FROM = {
					transfer_navy = { target = ROOT }
				}
			}
		}
	}

	on_declare_war = {
		effect = {
			# Great Patriotic War (event)
			if = {
				limit = {
					OR = {
						AND = {
							tag = RUS
							FROM = {
								OR = {
									tag = GER
									tag = FRA
								}
							}
						}
						AND = {
							OR = {
								tag = GER
								tag = FRA
							}
							FROM = { tag = RUS }
						}
					}
				}
				RUS = {
					country_event = {
						id = rusfor.47
						days = 3
					}
				}
			}
		}
	}

	on_new_term_election = {
		effect = {
			if = {
				limit = { tag = RUS }
				country_event = rusd.61
			}
			if = {
				limit = { tag = TRM }
				country_event = tram.87
			}
		}
	}

	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					ROOT = { tag = SOV }
					FROM = { tag = RUS }
				}
				ROOT = {
					country_event = { id = russia.1949 days = 1 random_hours = 30 }
				}
			}
			if = {
				limit = {
					ROOT = { tag = RUS }
					FROM = { tag = SOV }
				}
				ROOT = {
					country_event = { id = russia.1951 days = 1 random_hours = 30 }
				}
			}
		}
	}

	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			### A New 1812/1328 (event)
			if = {
				limit = {
					FROM = { tag = RUS }
					FROM.FROM = {
						OR = {
							state = 252
							state = 255
							state = 205
							state = 242
							state = 246
						}
					}
				}
				FROM = {
					country_event = {
						id = rusfor.51
						hours = 6
					}
				}
			}

			# Fall of Yekaterinburg (event)
			if = {
				limit = {
					tag = SOV
					FROM = { tag = RUS }
					FROM.FROM = { state = 653 } #Yekaterinburg
					ROOT = { controls_province = 6338 }
				}
				country_event = russia.1947
				FROM = { country_event = russia.1948 }
			}

			# RUS-FIN peace
			if = {
				limit = {
					OR = {
						is_russian_tag = yes
						tag = FIN
					}
					has_war_with = FROM
					FROM = {
						OR = {
							is_russian_tag = yes
							tag = FIN
						}
					}
				}
				FIN = { country_event = rusfor.1232 }
			}

			# RUS-ROM peace
			if = {
				limit = {
					OR = {
						is_russian_tag = yes
						tag = ROM
					}
					has_war_with = FROM
					FROM = {
						OR = {
							is_russian_tag = yes
							tag = ROM
						}
					}
				}
				ROM = { country_event = rusfor.1236 }
			}
		}
	}
}
