						tag = COS

					limit = {
						is_valid_peace_target = yes

							COS = { is_guaranteed_by = PREV }
							PAN = { is_guaranteed_by = PREV }

				country_event = { id = pru.106 days = 30 random_days = 120 }

						tag = NIC
