on_actions = {
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					NOT = { GXC = { has_completed_focus = GXC_chinas_gravest_hour } }
					OR = {
						AND = {
							tag = GXC
							FROM = {
								is_major = yes
								NOT = { tag = CUF }
							}
						}
						AND = {
							NOT = { tag = CUF }
							is_major = yes
							FROM = { tag = GXC }
						}
					}
				}
				GXC = { #if GXC at war with major, fire the focus
					complete_national_focus = GXC_chinas_gravest_hour
					mark_focus_tree_layout_dirty = yes
				}
			}
			if = {
				limit = {
					OR = {
						AND = {
							tag = GXC
							FROM = { tag = JAP }
						}
						AND = {
							tag = JAP
							FROM = { tag = GXC }
						}
					}
					GXC = {
						china_is_aligned_with_right_kmt = yes
						GXC_is_phase_3_trigger = yes
					}
				}
				GXC = { country_event = { id = GXC.midgame.499 days = 3 } } #Two suns in the sky
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = { tag = GXC }
				if = {
					limit = {
						FROM.FROM = { state = 608 } #Beijing
					}
					if = {
						limit = { china_is_aligned_with_right_kmt = yes }
						country_event = { id = GXC.midgame.498 days = 3 } #The Northern Expedition Avenged
					}
					else_if = {
						limit = { china_is_aligned_with_federalists = yes }
						country_event = { id = GXC.lategame.600 days = 3 } #Guangzhou Overcomes Beijing
					}
					else_if = {
						limit = {
							NOT = { GXC_is_phase_4_trigger = yes }
							NOT = {
								china_is_aligned_with_federalists = yes
								china_is_aligned_with_kmt = yes
							}
						}
						country_event = { id = GXC.lategame.800 days = 3 } #The March on Beijing
					}
				}
			}
		}
	}
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = { tag = GXC }
				#Hawaii's Prodigal Son - Note that Event Has an internal trigger so it won't fire if Phase III hasn't started. There is an alternative trigger if CHI dies before Phase III starts.
				if = {
					limit = {
						china_is_aligned_with_right_kmt = yes
						CHI = { exists = no }
					}
					country_event = { id = GXC.midgame.410 days = 30 } #Hawaii's Prodigal Son
				}
				if = {
					limit = {
						ROOT = { owns_state = 613 } #Nanjing
						china_is_aligned_with_right_kmt = yes
						CHI = { exists = no }
					}
					country_event = { id = GXC.midgame.520 days = 30 } #The Fall of the Left
					if = {
						limit = { ROOT = { NOT = { has_character = CHI_sun_fo } } }
						country_event = { id = GXC.midgame.410 days = 10 } #Hawaii's Prodigal Son
					}
				}
				if = {
					limit = {
						ROOT = { owns_state = 325 } #Kunming
						china_is_aligned_with_right_kmt = yes
						YUN = {
							OR = {
								exists = no
								AND = {
									exists = yes
									china_is_aligned_with_kmt = no
								}
							}
						}
					}
					country_event = { id = GXC.midgame.530 days = 7 } #Legacy of a Generalissimo
				}
				if = {
					limit = {
						ROOT = { owns_state = 602 } #Changsha
						china_is_aligned_with_right_kmt = yes
						HNN = {
							OR = {
								exists = no
								AND = {
									exists = yes
									china_is_aligned_with_kmt = no
								}
							}
						}
					}
					country_event = { id = GXC.midgame.531 days = 7 } #Matchstick of Revolution
				}
				if = {
					limit = {
						ROOT = { owns_state = 622 } #Shaanxi
						china_is_aligned_with_right_kmt = yes
						SHX = {
							OR = {
								exists = no
								AND = {
									exists = yes
									china_is_aligned_with_kmt = no
								}
							}
						}
					}
					country_event = { id = GXC.midgame.532 days = 7 } #The Estranged Northerners
				}
				if = {
					#On Peace Conference Ends applies to White Peaces as well thankfully
					limit = {
						has_country_flag = GXC_UPC_authoritarian_direction_flag
						FROM = { tag = JAP }
					}
					country_event = { id = GXC.lategame.692 days = 1 } #The 36 Points
				}
			}
		}
	}
	on_war = {
		effect = {
			if = {
				limit = {
					tag = GXC
					GXC_is_phase_4_trigger = yes
					has_country_flag = UPC_actively_holding_elections_flag
				}
				country_event = { id = GXC.lategame.606 days = 3 }  #A Wartime Political Truce
			}
		}
	}
	on_new_term_election = {
		effect = {
			if = {
				limit = {
					tag = UPC
					has_country_flag = UPC_has_drafted_federalist_constitution_flag
					NOT = { has_country_flag = UPC_has_federalist_democracy_flag }
				}
				country_event = UPC.federalelections.060 #The [GETYear] Elections
			}
		}
	}
}
