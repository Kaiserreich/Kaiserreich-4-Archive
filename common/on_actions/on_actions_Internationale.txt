on_actions = {
	#ROOT is winner #FROM is loser
	on_annex = {
		effect = {
			FROM = {
				FRA_remove_from_phalanstere = yes
				remove_from_tech_sharing_group = phalanstere_research
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation_immediate = {
		effect = {
			# France capitulates
			if = {
				limit = {
					tag = FRA
					NOT = { has_global_flag = FRA_capitulated }
				}
				country_event = { id = capt.40 days = 1 }
			}

			# Move the Internationale leader
			if = {
				limit = { tag = INT }
				### if SOV exists, merge the factions
				if = {
					limit = {
						SOV = { is_faction_leader = yes }
						NOT = { any_allied_country = { has_war_with = SOV } }
					}
					every_other_country = {
						limit = {
							is_in_faction_with = SOV
							is_subject = no
							NOT = { tag = SOV }
						}
						var:PREV.faction_leader = {
							add_to_faction = PREV
							hidden_effect = {
								diplomatic_relation = {
									country = PREV
									relation = guarantee
									active = no
								}
								diplomatic_relation = {
									country = PREV
									relation = non_aggression_pact
									active = no
								}
							}
						}
					}
					faction_leader = { add_to_faction = SOV }
					SOV = {
						set_faction_leader = yes
						save_global_event_target_as = KR_internationale_leader
					}
				}
				### collapse the Internationale if no true majors are left
				else_if = {
					limit = {
						NOT = {
							any_allied_country = {
								is_major = yes
								has_capitulated = no
								num_of_owned_factories > 50
							}
						}
					}
					every_other_country = {
						limit = {
							is_in_faction_with = ROOT
							is_subject = no
							has_capitulated = no
						}
						faction_collapse_member_peace_effect = yes
						country_event = comfrance.301
					}
					every_country = {
						FRA_remove_from_phalanstere = yes
						remove_from_tech_sharing_group = phalanstere_research
					}
					dismantle_faction = yes
					news_event = { id = comfrance.302 hours = 30 }
					if = {
						limit = { SOV = { is_faction_leader = yes } }
						SOV = { set_faction_name = fourth_internationale }
					}
					FRA = { save_global_event_target_as = KR_internationale_leader } #reset INT alias
				}
			}

			# SOV joins INT
			if = {
				limit = { tag = GER }
				INT = { country_event = { id = russoc.100 days = 2 } }
			}
		}
	}

	on_new_term_election = {
		effect = {
			if = {
				limit = { tag = FRA }
				country_event = comfrance.144
				country_event = comfrance.155
				country_event = comfrance.156
			}
		}
	}

	#FROM is country getting invited.
	on_offer_join_faction = {
		effect = {
			if = {
				limit = {
					tag = SOV
					is_faction_leader = yes
					INT = {
						is_faction_leader = yes
						NOT = {
							has_war_with = ROOT
							has_war_with = FROM
						}
					}
				}
				INT = {
					diplomatic_relation = {
						country = FROM
						relation = non_aggression_pact
					}
				}
			}
			else_if = {
				limit = {
					tag = INT
					SOV = {
						is_faction_leader = yes
						NOT = {
							has_war_with = ROOT
							has_war_with = FROM
						}
					}
				}
				SOV = {
					diplomatic_relation = {
						country = FROM
						relation = non_aggression_pact
					}
				}
			}
		}
	}

	#ROOT = attacking side
	#FROM = defending side
	#fired when two countries end up at war with each other (on_war is fired when a country goes to war against anyone and is not fired again when it enters war against another country unless it went to peace first)
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					FRA_is_phalanstere_member = yes
					INT = { is_ally_with = FROM }
				}
				FRA_remove_from_phalanstere = yes
			}
			else_if = {
				limit = {
					FROM = { FRA_is_phalanstere_member = yes }
					is_ally_with = INT
				}
				FROM = { FRA_remove_from_phalanstere = yes }
			}
		}
	}
}
