on_actions = {
	on_startup = {
		effect = {
			PHI = {
				set_temp_variable = { coalition_partner_var = token:social_conservative }
				add_to_coalition = yes
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
				limit = { original_tag = PHI }
				clr_country_flag = PHI_fight_for_the_philippines_active_flag
			}
		}
	}

	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				#Remove GEA from contention if they lose Singapore
				limit = {
					FROM = { tag = GEA }
					FROM.FROM = { state = 336 } #Singapore
					PHI = { has_country_flag = PHI_fight_for_the_philippines_active_flag }
				}
				PHI = { country_event = philippines_political_events.045 } #The Germans Depart From Manila
				GEA = { country_event = philippines_political_events.045 } #The Germans Depart From Manila
			}
			else_if = {
				#Remove JAP from contention if they lose Taiwan
				limit = {
					FROM = { tag = JAP }
					FROM.FROM = { state = 524 } #Taiwan
					PHI = { has_country_flag = PHI_fight_for_the_philippines_active_flag }
				}
				PHI = { country_event = philippines_political_events.046 } #The Japanese Depart From Manila
				JAP = { country_event = philippines_political_events.046 } #The Japanese Depart From Manila
			}
		}
	}
	#ROOT = attacking side
	#FROM = defending side
	#fired when two countries end up at war with each other (on_war is fired when a country goes to war against anyone and is not fired again when it enters war against another country unless it went to peace first)
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					ROOT = {
						OR = {
							tag = GEA
							tag = PHI
						}
					}
					FROM = {
						OR = {
							tag = GEA
							tag = PHI
						}
					}
				}
				PHI = { country_event = philippines_political_events.045 } #The Germans Pull Out of Manila
				GEA = { country_event = philippines_political_events.045 } #The Germans Pull Out of Manila
			}
			else_if = {
				limit = {
					ROOT = {
						OR = {
							tag = JAP
							tag = PHI
						}
					}
					FROM = {
						OR = {
							tag = JAP
							tag = PHI
						}
					}
				}
				PHI = { country_event = philippines_political_events.046 } #The Japanese Pull Out of Manila
				JAP = { country_event = philippines_political_events.046 } #The Japanese Pull Out of Manila
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation_immediate = {
		effect = {
			# logging
			if = {
				limit = { tag = PHI }
				if = {
					limit = { has_country_flag = PHI_PHI_won_fight_for_philippines_flag }
					log = "KR_Event_Logging;PHI ForPol - Independence, Death"
				}
				else_if = {
					limit = { has_country_flag = PHI_GER_won_fight_for_philippines_flag }
					log = "KR_Event_Logging;PHI ForPol - Germans, Death"
				}
				else_if = {
					limit = { has_country_flag = PHI_JAP_won_fight_for_philippines_flag }
					log = "KR_Event_Logging;PHI ForPol - Japanese, Death"
				}
				else_if = {
					limit = { has_country_flag = PHI_JAP_won_fight_for_philippines_flag }
					log = "KR_Event_Logging;PHI ForPol - Americans, Death"
				}
			}
		}
	}
}
