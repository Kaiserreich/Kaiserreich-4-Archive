		effect = {
			### Rebuilding the US Navy
			if = {
				limit = {
					is_american_tag = yes
					is_in_faction = no

				limit = {
					tag = HAW

			}

			if = {
				limit = {
					check_variable = { BRA.BRA_ACW_volunteers = ROOT }
				}

				limit = {
					ROOT = {
						NOT = {
							tag = USA
							tag = PSA
							tag = TEX

								tag = NEE

							}
						}

							tag = USA

					}
				}

					set_country_flag = ACW_attacked_during_recovery
					country_event = amcivwar.615
				}
			}
