on_actions = {
	on_unit_leader_level_up = {
		effect = {
			if = {
				limit = {
					FROM = { tag = HNN }
					ROOT = { has_trait = war_hero }
				}
				unit_leader_event = hunan_kuomintang_events.15 #Champion Event
			}
		}
	}
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					tag = HNN
					has_cosmetic_tag = HNN_SCPC
					is_subject = yes
				}
				#Southwest China Political Council Events
				#Restoring Order to Sichuan
				if = {
					limit = {
						owns_state = 605 #Kunming
						NOT = { country_exists = SZC }
					}
					country_event = { id = hunan_zhili_events.18 days = 7 }
				}
				#Taming the Republican Stronghold
				if = {
					limit = {
						owns_state = 325 #Kunming
						NOT = { country_exists = YUN }
					}
					country_event = { id = hunan_zhili_events.19 days = 7 }
				}
			}
			#Reclamation of Guizhou
			if = {
				limit = {
					tag = HNN
					china_is_aligned_with_zhili = yes
					owns_state = 1051 #Guiyang (capital of Guizhou)
					NOT = { country_exists = YUN }
				}
				country_event = { id = hunan_zhili_events.20 days = 14 }
			}
		}
	}
}
