on_actions = {
	on_startup = {
		effect = {
			### Set Up State Resistance/Compliance in SIA/BRM
			72 = { # Upper Shan
				set_resistance = 20
				set_compliance = 10
			}
			701 = { # Kachin
				set_resistance = 20
				set_compliance = 15
			}
			749 = { # Tanaosri
				set_resistance = 30
				set_compliance = 15
			}
			750 = { # Phnom Penh
				set_resistance = 5
				set_compliance = 10
			}
			751 = { # Trangkanu
				set_resistance = 25
				set_compliance = 10
			}
			756 = { # Kengtung
				set_resistance = 10
				set_compliance = 20
			}
			831 = { # Moulmein
				set_resistance = 35
				set_compliance = 10
			}
			882 = { # Chaiyaburi
				set_resistance = 5
				set_compliance = 30
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			if = {
				limit = {
					original_tag = SIA
					FROM = { original_tag = SIA }
				}
				country_event = siam.20 # Civil War ends
			}
		}
	}


	#ROOT is winner #FROM gets annexed - This fires just before FROM gets annexed, meaning the country and everything it owns still exists. It will also fire on_annex and on_civil_war_end
	on_civil_war_end_before_annexation = {
		effect = {
			### Rebuilding the Siamese Navy
			if = {
				limit = { original_tag = SIA }
				FROM = { transfer_navy = { target = ROOT } }
			}
		}
	}

	on_state_control_changed = {
		effect = {
			if = {
				### Fall of Bangkok
				limit = {
					original_tag = SIA
					FROM.FROM = { state = 289 } #Bangkok
				}
				country_event = { id = siam.18 days = 1 }
			}
			if = {
				## Remove SIA state reform if non-SIA get those states
				limit = {
					FROM.FROM = {
						OR = {
							state = 72
							state = 670
							state = 749
							state = 750
							state = 751
							state = 756
							state = 825
							state = 826
							state = 830
							state = 831
							state = 882
							state = 999
						}
					}
				}
				if = {
					limit = {
						FROM = { tag = SIA }
						FROM.FROM = { NOT = { is_core_of = FROM } }
					}
					FROM.FROM = {
						clr_state_flag = SIA_municipal_reform_implemented
						clr_state_flag = SIA_legal_reform_implemented
						clr_state_flag = SIA_civil_service_expansion_implemented
					}
				}
			}
		}
	}
}
