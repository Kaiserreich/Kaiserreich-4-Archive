on_actions = {
	#ROOT is winner #FROM is loser
	on_peaceconference_ended = {
		effect = {
			if = {
				limit = { tag = YUN }

				#Starts Digesting LKMT Event Chain
				if = {
					limit = {
						OR = {
							tag = RKT
							tag = LKT
						}
						CHI = { exists = no }
						owns_state = 613 #Nanjing
					}
					country_event = { id = YUN.midgame.440 days = 14 } #The White Sun Returns to Nanjing
				}

				if = {
					limit = {
						tag = RKT
						owns_state = 602 #Changsha
						HNN = { exists = no }
					}
					country_event = { id = YUN.midgame.581 days = 3 } #The Fall of Hunan
				}
			}
		}
	}

	#ROOT is new controller #FROM is old controller #FROM.FROM is state ID
	on_state_control_changed = {
		effect = {
			if = {
				limit = { tag = YUN }
				if = {
					limit = {
						tag = RKT
						FROM.FROM = {
							OR = {
								#These states are just above the Yellow River
								state = 1063 #Dezhou
								state = 1060 #Hebi
								state = 615 #Datong
								state = 1072 #Taiyuan
								state = 622 #Yulin
							}
						}
					}
					country_event = { id = YUN.midgame.582 days = 30 } #The Northern Kuomintang
				}
				if = {
					limit = {
						FROM.FROM = { state = 608 } #Beijing
					}
					if = {
						limit = { tag = RKT }
						country_event = { id = YUN.lategame.400 days = 3 } #The Northern Expedition Avenged
					}
					else_if = {
						limit = { tag = UPC }
						country_event = { id = YUN.lategame.600 days = 3 } #Cai Eâ€™s Crusade Completed
					}
				}
			}
		}
	}

	#FROM - Country (owner)
	#ROOT - leader
	#FROMFROM - original owner for exiled leaders
	on_unit_leader_level_up = {
		effect = {
			if = {
				limit = {
					FROM = { tag = YUN }
					ROOT = { has_trait = war_hero }
				}
				unit_leader_event = YUN.midgame.580 #Champion Event
			}
		}
	}
	#on war_relation_added
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					OR = {
						tag = RKT
						tag = LKT
					}
					FROM = { is_major = yes }
				}
				country_event = { id = YUN.lategame.490 days = 21 } #this fires for CHI and RKT
			}
			if = {
				limit = {
					NOT = { YUN = { has_completed_focus = YUN_the_nation_under_siege } }
					OR = {
						AND = {
							tag = YUN
							FROM = { is_major = yes }
						}
						AND = {
							is_major = yes
							FROM = { tag = YUN }
						}
					}
				}
				YUN = { #if YUN at war with major, fire the focus
					complete_national_focus = YUN_the_nation_under_siege
					mark_focus_tree_layout_dirty = yes
				}
			}
		}
	}

	#FROM is war target
	on_declare_war = {
		effect = {
			if = {
				limit = {
					tag = YUN
					OR = {
						YUN_is_phase_1_trigger = yes #Mostly for ease of testing, but basically want to check to make sure this isn't YUN doing it after forming a government.
						YUN_is_phase_2_trigger = yes
					}
					FROM = {
						is_chinese_or_potential_chinese_tag = yes
						owns_state = 594
					}
				}
				country_event = YUN.earlygame.801 #The Second Yunnan-Guangxi War
			}
		}
	}
}
