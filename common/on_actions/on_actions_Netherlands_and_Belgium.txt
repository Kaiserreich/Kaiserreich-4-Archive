on_actions = {
	on_startup = {
		effect = {

			#Coalitions
			HOL = {
				set_temp_variable = { coalition_partner_var = token:market_liberal }
				add_to_coalition = yes
			}
		}
	}

	# ROOT is capitulated country, FROM is winner
	on_capitulation_immediate = {
		effect = {
			### Dutch East Indies peace
			if = {
				limit = {
					HOL = { is_ally_with = ROOT }
					is_in_asia = yes
				}
				if = {
					limit = {
						BEL = {
							is_faction_leader = yes
							is_in_faction_with = HOL
							has_war_with = FROM
						}
					}
					BEL = {
						country_event = { id = east_asia_peace_events.4 days = 20 }
					}
				}
				else = {
					HOL = {
						country_event = { id = east_asia_peace_events.4 days = 20 }
					}
				}
			}
			if = {
				limit = { tag = INS }
				DEI = { set_country_flag = DEI_East_Indies_Rebellion_Crushed }
			}
		}
	}

	#ROOT is winner #FROM gets annexed - For civil wars on_civil_war_end is also fired
	on_annex = {
		effect = {
			FROM = {
				if = {
					limit = { is_belgian_tag = yes }
					drop_cosmetic_tag = yes
				}
				else_if = {
					limit = {
						tag = HOL
						DEI = {
							exists = yes
							OR = {
								is_subject = no
								is_subject_of = HOL
							}
						}
					}
					HOL_set_up_dutch_exiles = yes
				}
			}
		}
	}

	on_peaceconference_ended = {
		effect = {
			if = {
				limit = { original_tag = GER }
				country_event = { id = belgium_independent_events.56 days = 1 }
				INT = { country_event = { id = belgium_independent_events.56 days = 1 } }
			}
			else_if = {
				limit = { FROM = { original_tag = GER } }
				INT = { country_event = { id = belgium_independent_events.56 days = 1 } }
			}
		}
	}

	on_ruling_party_change = {
		effect = {
			if = {
				limit = { tag = BEL }
				if = {
					limit = {
						has_idea = MAF_African_Investments
						has_socialist_government = yes
					}
					remove_ideas = MAF_African_Investments
				}
			}
			else_if = {
				limit = { tag = HOL }
				if = {
					limit = {
						has_socialist_government = yes
						previous_government_was_socialist = no
					}
					HOL_add_socialist_ideas = yes
					remove_ideas = {
						HOL_idea_ambtenarenverbod
						HOL_idea_ambtenarenverbod_01
						HOL_idea_ambtenarenverbod_02
						HOL_idea_orde_van_oranje_nassau
						HOL_idea_outlawed_unions
						HOL_idea_agreement_with_left_0
						HOL_idea_agreement_with_left_1
						HOL_idea_agreement_with_left_2
						HOL_idea_agreement_with_right_1
						HOL_idea_agreement_with_right_2
					}
					country_event = { id = netherlands_post_election_events.122 days = 4 }
					GER = { country_event = { id = netherlands_post_election_events.117 days = 7 } }
				}
				else = {
					HOL_remove_socialist_ideas = yes
				}
				if = {
					limit = {
						has_socialist_government = yes
						NOT = { has_country_flag = HOL_disabled_indonesian_ambition_areas }
					}
					set_country_flag = HOL_disabled_indonesian_ambition_areas
					every_state = {
						limit = {
							OR = {
								region = 158 #Sunda Islands
								region = 192 #Sumatra
								region = 93 #Java & Flores Seas
								region = 159 #Borneo
								region = 167 #New Guinea
								region = 91 #Arafura & Banda Seas
								region = 99 #Indian Ocean
								region = 71 #Indian Ocean
							}
						}
						clear_variable = ambition_area_HOL
					}
				}
				else_if = {
					limit = {
						has_country_flag = HOL_disabled_indonesian_ambition_areas
						has_socialist_government = no
					}
					clr_country_flag = HOL_disabled_indonesian_ambition_areas
					every_state = {
						limit = {
							OR = {
								region = 158 #Sunda Islands
								region = 192 #Sumatra
								region = 93 #Java & Flores Seas
								region = 159 #Borneo
								region = 167 #New Guinea
								region = 91 #Arafura & Banda Seas
								region = 99 #Indian Ocean
								region = 71 #Indian Ocean
							}
						}
						set_variable = { ambition_area_@HOL = 1 }
					}
				}
			}
		}
	}

	on_new_term_election = {
		effect = {
			### Belgium ###
			if = {
				limit = { tag = BEL }
				country_event = belgium_events.1000
				country_event = belgium_independent_events.14
			}
			### Flanders ###
			else_if = {
				limit = { tag = FLA }
				country_event = flanders_events.101 #Flemish Generic
			}
			### Netherlands ###
			else_if = {
				limit = { tag = HOL }
				country_event = netherlands_election_events.1 #1937
				country_event = netherlands_post_election_events.70 #Oud 1941
				country_event = netherlands_post_election_events.76 #Colijn 1941
				country_event = netherlands_post_election_events.81 #Swart 1941
				country_event = netherlands_post_election_events.85 #Republican
				country_event = netherlands_post_election_events.105 #Reijnders 1945+
				country_event = netherlands_post_election_events.106 #VDB/RSAP 1945+
				country_event = netherlands_post_election_events.111 #CNA 1945+
			}
			### Wallonia ###
			else_if = {
				limit = { tag = WAL }
				country_event = wallonia_events.718
			}
		}
	}

	on_offer_join_faction = {
		effect = {
			if = {
				limit = { tag = HOL }
				remove_ideas = {
					HOL_idea_ambtenarenverbod
					HOL_idea_ambtenarenverbod_01
					HOL_idea_ambtenarenverbod_02
					HOL_idea_orde_van_oranje_nassau
					HOL_idea_outlawed_unions
					HOL_idea_agreement_with_left_0
					HOL_idea_agreement_with_left_1
					HOL_idea_agreement_with_left_2
					HOL_idea_agreement_with_right_1
					HOL_idea_agreement_with_right_2
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
			FROM.FROM = {
				if = {
					limit = {
						state = 980
						has_state_flag = FLA_conscription_flag
					}
					if = {
						limit = { ROOT = { tag = FLA } }
						if = {
							limit = { NOT = { has_dynamic_modifier = { modifier = FLA_integration_of_french_flanders } } }
							add_dynamic_modifier = { modifier = FLA_integration_of_french_flanders }
						}
					}
					else = {
						if = {
							limit = { has_dynamic_modifier = { modifier = FLA_integration_of_french_flanders } }
							remove_dynamic_modifier = { modifier = FLA_integration_of_french_flanders }
						}
					}
				}
			}
		}
	}

	on_war_relation_added = {
		effect = {
			if = {
				limit = { #if INT and GER go to war, and BEL has neutrality
					ROOT = {
						OR = {
							original_tag = GER
							tag = INT
						}
					}
					FROM = {
						OR = {
							original_tag = GER
							tag = INT
						}
					}
					BEL = {
						exists = yes
						OR = {
							is_faction_leader = yes
							is_in_faction = no
						}
					}
				}
				if = {
					limit = {
						OR = {
							GER = {
								has_authoritarian_government = yes
							}
							AND = {
								GER = { has_authoritarian_government = no }
								INT = { NOT = { has_government = totalist } }
							}
						}
					}
					GER = { country_event = { id = belgium_independent_events.55 days = 1 } }
				}
				else = {
					FRA = { country_event = { id = belgium_independent_events.52 days = 1 } }
				}
			}
			if = {
				limit = {
					tag = GER
					is_valid_peace_target = yes
					FROM = {
						original_tag = BEL
						has_socialist_government = no
						is_valid_peace_target = yes
					}
				}
				FROM = { activate_mission = BEL_peace_with_germany }
			}
			else_if = {
				limit = {
					tag = BEL
					has_socialist_government = no
					is_valid_peace_target = yes
					FROM = {
						original_tag = GER
						is_valid_peace_target = yes
					}
				}
				activate_mission = BEL_peace_with_germany
			}
		}
	}

	on_weekly_BEL = {
		effect = {
			GER = { country_event = belgium_events.73 }
		}
	}
}
