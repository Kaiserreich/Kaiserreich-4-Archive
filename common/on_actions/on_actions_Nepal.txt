on_actions = {
	on_ruling_party_change_immediate = {
		effect = {
			if = {
				limit = { original_tag = NEP }
				if = {
					limit = { is_monarchy = yes }
					if = {
						limit = { has_government = national_populist }
						activate_advisor = NEP_purendra_bikram_shah_sic
					}
					else_if = {
						limit = { has_government = paternal_autocrat }
						activate_advisor = NEP_juddha_shumsher_jang_bahadur_rana_sic
					}
					else_if = {
						limit = { has_elected_government = yes }
						activate_advisor = NEP_tanka_prasad_acharya_sic
					}
					else = {
						activate_advisor = NEP_matrika_prasad_koirala_sic
					}
					NEP_army_council = {
						set_character_name = NEP_army_council
					}
				}
				else = {
					if = {
						limit = { has_socialist_government = yes }
						activate_advisor = NEP_kunwar_inderjit_singh_sic
					}
					else_if = {
						limit = { has_dictatorship_government = yes }
						activate_advisor = NEP_kaiser_shumsher_jang_bahadur_rana_sic
					}
					else_if = {
						limit = { NEP_dashrath_chand = { is_dead = no } }
						activate_advisor = NEP_dashrath_chand_sic
					}
					else = {
						activate_advisor = NEP_dharma_bhakta_mathema_sic
					}
					NEP_army_council = {
						set_character_name = NEP_army_council_republican
					}
				}
			}
		}
	}
}
