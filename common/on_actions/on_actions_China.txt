				}

					limit = { has_idea = GXW_lost_momentum }
					remove_ideas = GXW_lost_momentum

						limit = { has_active_mission = GXC_move_quickly }
						remove_mission = GXC_move_quickly

						limit = { has_idea = GXC_victory_imminent }
						remove_ideas = GXC_victory_imminent
					}
					if = {
						limit = { has_idea = GXC_lost_momentum }
						remove_ideas = GXC_lost_momentum
					}
					if = {
						limit = { has_idea = GXW_victory_imminent }
						remove_ideas = GXW_victory_imminent
					}
					if = {
						limit = { has_idea = GXW_lost_momentum }
						remove_ideas = GXW_lost_momentum
					}
					every_character = {
						limit = {
							OR = {
								has_trait = GXC_guangdong_clique_officer
								has_trait = GXC_guangxi_clique_officer
								is_advisor = yes
							}
						}
						if = {
							limit = { can_be_country_leader = no }
							set_nationality = ROOT
						}
						else = {
							retire = yes
						}
					}
				}
				if = {
					limit = { GXC_has_guangdong_government = yes }
					if = {
						limit = { has_global_flag = GXC_new_guangxi_clique_took_over }
						GXC_new_guangxi_clique_flees = yes
					}
					else = {
						GXC_old_guangxi_clique_flees = yes
					}
				}

					limit = { GXC_has_federalist_government = no }
					GXC_federalists_flee = yes
				}
				clear_global_event_target = GXC_civil_war_target
				clear_global_event_target = GXC_original_target
			}
		}
	}
