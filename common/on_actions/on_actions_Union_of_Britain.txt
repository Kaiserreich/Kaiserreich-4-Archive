on_actions = {
	on_declare_war = {
		effect = {
			if = {
				limit = {
					tag = CAN
					is_faction_leader = yes
					has_global_flag = ENG_CANIRE_protection
					ENG = {
						has_war_with = IRE
						has_socialist_government = yes
					}
					NOT = {
						has_socialist_government = yes
						has_war_with = ENG
					}
					IRE = {
						is_in_faction = no
						is_subject = no
						NOT = {
							is_guaranteed_by = CAN
							is_guaranteed_by = GER
						}
					}
				}
				country_event = { id = uob.31 days = 1 }
			}
			if = {
				limit = {
					tag = GER
					is_faction_leader = yes
					has_global_flag = ENG_GERIRE_protection
					ENG = {
						has_socialist_government = yes
						has_war_with = IRE
					}
					NOT = {
						has_socialist_government = yes
						has_war_with = ENG
					}
					IRE = {
						is_in_faction = no
						is_subject = no
						NOT = {
							is_guaranteed_by = CAN
							is_guaranteed_by = GER
						}
					}
				}
				country_event = { id = uob.37 days = 1 }
			}
			if = {
				limit = {
					tag = CAN
					is_faction_leader = yes
					has_global_flag = ENG_CANIRE_support
					ENG = {
						has_war_with = IRE
						has_socialist_government = yes
					}
					NOT = {
						has_socialist_government = yes
						has_war_with = ENG
					}
					IRE = {
						is_in_faction = no
						is_subject = no
						NOT = {
							is_guaranteed_by = CAN
							is_guaranteed_by = GER
						}
					}
				}
				country_event = { id = uob.43 days = 1 }
			}
			if = {
				limit = {
					tag = GER
					is_faction_leader = yes
					has_global_flag = ENG_GERIRE_support
					ENG = {
						has_socialist_government = yes
						has_war_with = IRE
					}
					NOT = {
						has_socialist_government = yes
						has_war_with = ENG
					}
					IRE = {
						is_in_faction = no
						is_subject = no
						NOT = {
							is_guaranteed_by = CAN
							is_guaranteed_by = GER
						}
					}
				}
				country_event = { id = uob.44 days = 1 }
			}
			if = {
				limit = {
					tag = IRE
					has_idea = IRE_LAST_STAND
					has_war_with = ENG
					is_in_faction = no
					is_subject = no
					NOT = {
						has_war_with = GER
						has_war_with = CAN
					}
					ENG = {
						has_war_with = GER
						surrender_progress < 0.20
					}
					GER = {
						is_faction_leader = yes
					}
				}
				country_event = { id = uob.76 days = 7 }
			}
			else_if = {
				limit = {
					tag = IRE
					has_idea = IRE_LAST_STAND
					has_war_with = ENG
					is_in_faction = no
					is_subject = no
					NOT = {
						has_war_with = CAN
						has_war_with = GER
					}
					ENG = {
						has_war_with = CAN
						surrender_progress < 0.20
					}
					CAN = {
						is_faction_leader = yes
					}
				}
				country_event = { id = uob.80 days = 14 }
			}
		}
	}

	#ROOT is winner, FROM is loser (called for all winners against all losers)
	on_peaceconference_started = {
		effect = {
			#store ENG factory counts
			if = {
				limit = {
					FROM = {
						original_tag = ENG
						has_capitulated = yes
					}
				}
				FROM = { every_core_state = { save_current_factory_count = yes } }
			}
		}
	}

	on_peaceconference_ended = {
		effect = {
			if = {
				limit = {
					is_english_tag = yes
					has_country_flag = ENG_irish_uprising
					OR = {
						AND = {
							113 = { is_owned_and_controlled_by = ROOT }
							IRE = { exists = no }
						}
						AND = {
							country_exists = IRE
							IRE = { is_subject_of = ROOT }
						}
					}
				}
				country_event = { id = uob.98 days = 10 }
			}
			if = {
				limit = {
					is_english_tag = yes
					has_country_flag = ENG_IRA
					country_exists = IRE
					NOT = {
						controls_state = 113
						has_war_with = IRE
					}
				}
				country_event = { id = uob.85 days = 10 }
			}
		}
	}
	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					is_english_tag = yes
					has_country_flag = ENG_IRA
					country_exists = IRE
					NOT = {
						controls_state = 113
						has_war_with = IRE
					}
				}
				country_event = { id = uob.85 days = 10 }
			}
		}
	}
}
