	allowed = {
		original_tag = FRA
	}
	enable = {
		NOT = { has_government = totalist }
		OR = {
			BEL = {
				is_faction_leader = yes
				has_idea = generic_neutral_foreign_policy
			}
			HOL = {
				is_faction_leader = yes
				has_idea = generic_neutral_foreign_policy
			}
		}
		OR = {
			has_war = yes
			GER = {
				exists = yes
				is_subject = no
				has_socialist_government = no
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = { type = ignore_claim id = BEL value = 1 }
