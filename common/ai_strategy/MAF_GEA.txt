# Keep MAF units in Africa
KR_keep_MAF_units_home = {
	reversed = yes

	enable = {
		NOT = { tag = MAF }
		NOT = { has_global_flag = second_weltkrieg_has_started }
		is_in_faction_with = MAF
		MAF = { is_subject_of = GER }
	}

	abort = {
		OR = {
			has_global_flag = second_weltkrieg_has_started
			NOT = { is_in_faction_with = MAF }
			NOT = { MAF = { is_subject_of = GER } }
		}
	}

	ai_strategy = {
		type = dont_defend_ally_borders
		id = "MAF"
		value = 100
	}
}

KR_MAF_prewar_strategy = {
	enable = {
		tag = MAF
		NOT = { has_global_flag = second_weltkrieg_has_started }
		is_subject_of = GER
	}

	abort = {
		OR = {
			has_global_flag = second_weltkrieg_has_started
			NOT = { is_subject_of = GER }
		}
	}

	ai_strategy = {
		type = area_priority
		id = benelux
		value = -1000
	}
	ai_strategy = {
		type = area_priority
		id = germany
		value = -1000
	}
	ai_strategy = {
		type = area_priority
		id = africa
		value = 500
	}
}

# Navy won't die easily to Japan
KR_GEA_war_strategy = {
	enable = {
		tag = GEA
		has_war_with = JAP
		OR = {
			naval_strength_ratio = { tag = JAP ratio < 0.75 }
			country_exists = FRA #France is not defeated
		}
	}
	abort = {
		OR = {
			NOT = { has_war_with = JAP }
			naval_strength_ratio = { tag = JAP ratio > 0.75 }
			NOT = { country_exists = FRA } #France is defeated
		}
	}
	ai_strategy = {
		type = invade #Not even try islands
		id = "JAP"
		value = -1500
	}
	ai_strategy = {
		type = naval_avoid_region
		id = 79 #Sea of Japan
		value = 100
	}
	ai_strategy = {
		type = naval_avoid_region
		id = 90 #Coast of Japan
		value = 100
	}
	ai_strategy = {
		type = naval_avoid_region
		id = 76 #East China Sea
		value = 100
	}
	ai_strategy = {
		type = naval_avoid_region
		id = 77 #Yellow Sea
		value = 100
	}
	ai_strategy = {
		type = area_priority
		id = east_indies
		value = 1000
	}
	ai_strategy = {
		type = area_priority
		id = se_asia
		value = 1000
	}
}
