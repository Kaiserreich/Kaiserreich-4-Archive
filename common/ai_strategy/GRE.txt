GRE_put_units_on_BUL_border = {
	allowed = {
		original_tag = GRE
	}
	enable = {
		OR = {
			is_in_faction_with = SER
			AND = {
				is_in_faction = no
				has_war = no
			}
			has_war_with = TUR
		}
		OR = {
			AND = {
				SER = { focus_progress = { focus = SER_ultimatum_to_bulgaria progress > 0 } }
				NOT = { SER = { has_completed_focus = SER_ultimatum_to_bulgaria } }
			}
			AND = {
				has_war_with = TUR
				BUL = { focus_progress = { focus = BUL_cairo_pact_strike progress > 0 } }
				NOT = { BUL = { has_completed_focus = BUL_cairo_pact_strike } }
			}
			AND = {
				BUL = { has_war_with = TUR }
				GRE = { has_country_flag = GRE_about_to_attack_TUR }
			}
			AND = {
				country_exists = MAC
				has_war = no
			}
		}
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = conquer
		id = BUL
		value = 400
	}
	ai_strategy = {
		type = front_unit_request
		tag = BUL
		value = 100
	}
	ai_strategy = {
		type = conquer
		id = MAC
		value = 400
	}
	ai_strategy = {
		type = front_unit_request
		tag = MAC
		value = 100
	}
	ai_strategy = {
		type = dont_defend_ally_borders
		id = SER
		value = 1000
	}
	ai_strategy = {
		type = front_unit_request
		country_trigger = {
			NOT = {
				original_tag = BUL
				original_tag = MAC
			}
		}
		value = -400
	}
