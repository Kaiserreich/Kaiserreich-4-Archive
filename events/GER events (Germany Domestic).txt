# Cosmetic Change for GEX
country_event = {
	id = gerfla.11
	title = gerfla.11.t
	desc = gerfla.11.d
	picture = GFX_report_event_german_reichstag_gathering

	fire_only_once = yes

	trigger = {
		tag = GEX
		NOT = { has_cosmetic_tag = MAF_germany }
		OR = {
			64 = { is_owned_and_controlled_by = ROOT } #Brandenburg - Berlin
			55 = { is_owned_and_controlled_by = ROOT } #Hessen - Frankfurt
			59 = { is_owned_and_controlled_by = ROOT } #Hannover - Hamburg
			51 = { is_owned_and_controlled_by = ROOT } #Ostrheinland - Essen
			52 = { is_owned_and_controlled_by = ROOT } #Upper Bavaria - Munich
		}
	}

	mean_time_to_happen = {
		days = 3
	}

	option = {
		log = "[GetLogInfo]: event gerfla.11 option gerfla.11.a"
		name = gerfla.11.a
		remove_ideas = MAF_on_borrowed_time
		add_political_power = 50
		add_stability = 0.2
		set_cosmetic_tag = MAF_germany
		if = {
			limit = {
				64 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = { state = 64 remember_old_capital = no }
		}
		else_if = {
			limit = {
				55 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = { state = 55 remember_old_capital = no }
		}
		else_if = {
			limit = {
				59 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = { state = 59 remember_old_capital = no }
		}
		else_if = {
			limit = {
				51 = {
					is_owned_and_controlled_by = ROOT
				}
			}
			set_capital = { state = 51 remember_old_capital = no }
		}
		else = {
			set_capital = { state = 52 remember_old_capital = no }
		}
		if = {
			limit = { is_in_array = { global.african_tags_array = ROOT } }
			remove_from_array = { global.african_tags_array = ROOT }
			add_to_array = { global.european_tags_array = ROOT }
		}
		every_state = {
			limit = {
				is_core_of = ROOT
				is_on_continent = africa
			}
			remove_core_of = ROOT
			add_claim_by = ROOT
		}
	}
}
