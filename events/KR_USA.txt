			has_war = yes

			is_in_faction_with = ENG

			is_in_faction_with = RUS
		}
	}

	mean_time_to_happen = {
		days = 35
	}

	option = {
		name = ustates.197.a

			factor = 70
		}
		trigger = { country_exists = CAN CAN = { is_subject = no is_faction_leader = yes } }

			limit = { NOT = { CAN = { is_in_faction_with = GBR }}}

		}

		add_political_power = -40
		clr_global_flag = FSA_LIB
		add_opinion_modifier = { target = CAN modifier = FSA_JOINS_ENTENTE }
	}

	option = {
		name = ustates.197.b

		ai_chance = {
			factor = 50
		}

	}

	option = {
		name = ustates.197.c

		ai_chance = {
			factor = 10
		}
		trigger = { country_exists = RUS NOT = { country_exists = DKB country_exists = SOV } RUS = { has_war = no controls_state = 219 controls_state = 234 controls_state = 566 controls_state = 562 is_subject = no NOT = { has_country_flag = RUS_BOL is_in_faction_with = JAP is_in_faction_with = FRA is_in_faction_with = CAN is_in_faction_with = GER } } }

		add_political_power = -40
		clr_global_flag = FSA_LIB
		add_opinion_modifier = { target = RUS modifier = FSA_JOINS_RUSS }
	}

	option = {
		name = ustates.197.e

		ai_chance = {
			factor = 15
		}

		add_political_power = 20
	}
