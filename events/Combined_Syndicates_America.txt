	immediate = {log = "[GetDateText]: [Root.GetName]: event csa.129"}
	title = csa.129.t
	desc = csa.129.d
	picture = congress

	fire_only_once = yes

	trigger = {
		tag = CSA
		NOT = {
			country_exists = TEX
			country_exists = USA
			country_exists = CAL
			country_exists = NEE
		}
		owns_state = 361 #Maryland
		owns_state = 371 #Louisiana
		owns_state = 395 #Illinois
		owns_state = 797 #San Francisco
		owns_state = 932 #Massachusetts
		controls_all_owned_states = yes
	}

	option = { # at long last we are reunited
		name = csa.129.a
		set_country_flag = CSA_america_united
		every_state = {
			remove_ACW_cores = yes
		}
	}
}

country_event = { # Overtures to Mexico
	id = csa.130
	immediate = {log = "[GetDateText]: [Root.GetName]: event csa.130"}
	title = csa.130.t
	desc = csa.130.d
	picture = diplomacy2

	is_triggered_only = yes

	option = {
		name = csa.130.a #Agree
		ai_chance = {
			factor = 90
			modifier = {
				factor = 0.5
				has_opinion = {
					target = FROM
					value < 0
				}
			}
		}
		add_stability = 0.05
		add_political_power = 100
		reverse_add_opinion_modifier = {
			target = FROM
			modifier = USA_american_trade
		}
		reverse_add_opinion_modifier = {
			target = FROM
			modifier = ITA_likes_govt
		}
		if = {
			limit = {
				is_subject = no
				NOT = {
					is_guaranteed_by = FROM
				}
			}
			FROM = {
				give_guarantee = MEX
			}
		}
		if = {
			limit = {
				NOT = {
					gives_military_access_to = FROM
				}
			}
			give_military_access = FROM
		}
		every_owned_state = {
			limit = {
				is_mainland_america = yes
				NOT = {
					is_core_of = MEX
				}
			}
			if = {
				limit = {
					is_claimed_by = MEX
				}
				remove_claim_by = MEX
			}
			CSA = {
				transfer_state = PREV
			}
		}
		FROM = {
			every_owned_state = {
				limit = {
					NOT = {
						is_core_of = FROM
						is_claimed_by = FROM
					}
					is_mainland_america = no
					is_core_of = MEX
				}
				MEX = {
					transfer_state = PREV
				}
			}
			country_event = {
				id = csa.131
				hours = 8
			}
		}
	}
	option = {
		name = csa.130.b #Refuse
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				OR = {
					is_subject_of = FROM
					is_in_faction_with = FROM
					has_opinion = {
						target = FROM
						value > 50
					}
				}
			}
		}
		add_stability = -0.05
		add_political_power = -50
		reverse_add_opinion_modifier = {
			target = FROM
			modifier = ITA_dislikes_govt
		}
		FROM = {
			country_event = {
				id = csa.132
				hours = 8
			}
		}
	}
}

country_event = { # Longist terrorism - factory sabotaged
	id = csa.200
	immediate = {log = "[GetDateText]: [Root.GetName]: event csa.200"}
	title = csa.200.t
	desc = csa.200.d
	picture = GFX_report_event_SRI_bombing

	trigger = {
		tag = CSA
		OR = {
			has_idea = USA_longist_resistance1
			has_idea = USA_longist_resistance2
			has_idea = USA_longist_resistance3
			has_idea = USA_longist_resistance4
		}
		any_owned_state = {
			OR = {
				is_core_of = USA
				is_core_of = CSA
				is_core_of = TEX
				is_core_of = CAL
			}
			industrial_complex > 0
			is_controlled_by = CSA
		}
	}

	mean_time_to_happen = {
		days = 70
		modifier = {
			factor = 1.25
			has_idea = USA_longist_resistance2
		}
		modifier = {
			factor = 1.5
			has_idea = USA_longist_resistance3
		}
		modifier = {
			factor = 2
			has_idea = USA_longist_resistance4
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_state = {
				limit = {
					OR = {
						is_core_of = USA
						is_core_of = CSA
						is_core_of = TEX
						is_core_of = CAL
					}
					industrial_complex > 0
					is_controlled_by = CSA
				}
				save_event_target_as = attack_state
			}
		}
	}

	option = {
		name = csa.200.a
		add_stability = -0.01
		event_target:attack_state = {
			damage_building = {
				type = industrial_complex
				damage = 1.0
			}
			if = {
				limit = {
					arms_factory > 0
				}
				random_list = {
					50 = {
						event_target:attack_state = {
							damage_building = {
								type = arms_factory
								damage = 1.0
							}
						}
					}
					50 = {}
				}
			}
			if = {
				limit = {
					infrastructure > 3
				}
				random_list = {
					25 = {
						event_target:attack_state = {
							damage_building = {
								type = infrastructure
								damage = 2.0
							}
						}
					}
					50 = {
						event_target:attack_state = {
							damage_building = {
								type = infrastructure
								damage = 1.0
							}
						}
					}
					25 = {}
				}
			}
			random_list = {
				25 = {
					ROOT = {
						add_manpower = -3000
					}
				}
				50 = {
					ROOT = {
						add_manpower = -1500
					}
				}
				25 = {
					ROOT = {
						add_manpower = -500
					}
				}
			}
		}
	}
}
