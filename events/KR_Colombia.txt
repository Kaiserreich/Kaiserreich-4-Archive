country_event = {

	id = col.5
	title = col.5.t
	desc = col.5.d
	picture = GFX_report_event_german_troops

	is_triggered_only = yes

	trigger = {
		NOT = {
			has_war_with = COL
		}
	}

	option = {
		name = col.5.a
		ai_chance = {
			factor = 20
		}
		COL = {
			country_event = {
				id = col.6 days = 1
			}
		}
	}

	option = {
		name = col.5.b
		ai_chance = {
			factor = 80
			modifier = {
				factor = 3
				strength_ratio = {
					tag = COL
					ratio > 0.7
				}
			}
			modifier = {
				factor = 0.1
				any_other_country = {
					is_major = yes
					is_in_faction_with = ECU
				}
			}
		}
		COL = {
			country_event = {
				id = col.7 days = 1
			}
			if = {
				limit = {
					is_in_faction_with = ECU
					is_faction_leader = yes
				}
				remove_from_faction = ECU
			}
			if = {
				limit = {
					is_in_faction_with = ECU
					NOT = { is_faction_leader = yes }
				}
				random_other_country = {
					limit = {
						is_in_faction_with = COL
						is_faction_leader = yes
					}
					remove_from_faction = COL
				}
			}
		}
		effect_tooltip = {
			COL = {
				declare_war_on = {
					target = ECU
					type = annex_everything
				}
			}
		}
	}
}
