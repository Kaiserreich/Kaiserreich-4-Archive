country_event = { #Fate of America
	id = annex.2000
	title = annexation_USA_CONQ
	desc = annexation_USA_CONQ_desc
	picture = GFX_report_event_generic_conference2

	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			add_to_temp_array = { core_removal_array = USA }
			add_to_temp_array = { core_removal_array = CSA }
			add_to_temp_array = { core_removal_array = PSA }
			add_to_temp_array = { core_removal_array = TEX }
			remove_all_cores_and_claims_of_tags = yes
			if = {
				limit = { is_russian_tag = yes }
				add_state_claim = 463
			}
			every_state = {
				limit = {
					OR = {
						is_mainland_america = yes
						state = 463 #Alaska
					}
				}
				add_core_of = USA
			}
			every_state = {
				limit = { is_american_pacific_islands = yes }
				add_core_of = HAW
			}
			CSA = {
				if = {
					limit = { has_character = CSA_jack_reed }
					retire_character = CSA_jack_reed
				}
			}
			USA = {
				if = {
					limit = { NOT = { has_character = USA_douglas_macarthur } }
					USA_douglas_macarthur = { set_nationality = USA }
					promote_character = { character = USA_douglas_macarthur ideology = paternal_autocrat_subtype }
				}
			}
		}
	}

	option = { #Liberate NEE as USA (Entente option)
		log = "[GetLogInfo]: event annex.2000 option annex.2000.a"
		name = annex.2000.a
		trigger = {
			OR = {
				tag = ENT
				is_in_faction_with = ENT
			}
			has_socialist_government = no
		}
		ai_chance = { base = 60 }
		set_temp_variable = { tag_to_release = NEE }
		hidden_effect = {
			NEE = {
				set_cosmetic_tag = america
				# Set flags
				set_country_flag = NEE_claimed_america
				set_country_flag = NEE_focusing_Entente
				NEE_united_states_congress = { set_character_name = USA_united_states_congress }
				every_state = {
					limit = {
						NOT = { is_claim_or_core_of_ROOT = yes }
						OR = {
							is_mainland_america = yes
							state = 463 #Alaska
						}
					}
					add_core_of = NEE
				}
			}
			release_targeted_tag_nocosmetic = yes
			NEE = {
				meta_effect = {
					text = {
						set_politics = {
							ruling_party = market_liberal
							last_election = "[CURRENT_DATE]"
							election_frequency = 1
							elections_allowed = yes
						}
					}
					CURRENT_DATE = "[?global.date.GetDateStringNoHour]"
				}
				set_popularities = {
					totalist = 0
					radical_socialist = 4
					syndicalist = 9
					social_democrat = 5
					social_liberal = 28
					market_liberal = 33
					social_conservative = 7
					authoritarian_democrat = 9
					paternal_autocrat = 2
					national_populist = 3
				}
			}
		}
		effect_tooltip = {
			puppet = USA
		}
		recheck_annexations = yes
	}

	option = { #Liberate USA
		log = "[GetLogInfo]: event annex.2000 option annex.2000.a"
		name = annex.2000.a
		trigger = {
			NOT = {
				tag = ENT
				is_in_faction_with = ENT
			}
			has_socialist_government = no
		}
		ai_chance = { base = 60 }
		set_temp_variable = { tag_to_release = USA }
		release_targeted_tag_nocosmetic = yes
		hidden_effect = {
			USA = {
				if = {
					limit = {
						has_government = market_liberal
						has_country_leader = { character = USA_douglas_macarthur }
					}
					random_list = {
						1 = { promote_character = USA_alf_landon }
						1 = { promote_character = USA_wendell_willkie }
					}
				}
			}
		}
		recheck_annexations = yes
	}

	option = { #Liberate CSA
		log = "[GetLogInfo]: event annex.2000 option annex.2000.b"
		name = annex.2000.b
		trigger = {
			has_socialist_government = yes
		}
		ai_chance = { base = 60 }
		set_temp_variable = { tag_to_release = CSA }
		every_state = {
			limit = {
				OR = {
					is_mainland_america = yes
					state = 463 #Alaska
				}
			}
			remove_core_of = USA
			add_core_of = CSA
		}
		release_targeted_tag = yes
		var:tag_to_release = {
			unlock_national_focus = CSA_second_convention
			if = {
				limit = { has_government = totalist }
				complete_national_focus = CSA_usas
			}
			else = {
				complete_national_focus = CSA_american_commonwealth
			}
		}
		recheck_annexations = yes
	}

	option = { #Occupy USA
		log = "[GetLogInfo]: event annex.2000 option annex.occupation"
		name = annex.occupation
		ai_chance = { base = 0 }
		occupy_massive_country = yes
		set_country_flag = USA_CONQ
		recheck_annexations = yes
	}
}
