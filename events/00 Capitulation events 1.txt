# Main German capitulation event
country_event = {
	id = capitulation_events.1
	title = capitulation_events.1.t
	desc = capitulation_events.1.d
	picture = GFX_report_event_german_reichstag_bombed

	fire_only_once = yes

	trigger = {
		tag = GER
		is_in_faction = yes
		is_subject = no
		OR = {
			has_event_target = GER_manual_collapse_triggered
			AND = {
				surrender_progress > 0.8
				NOT = {
					any_allied_country = {
						is_major = yes
						is_subject = no
						is_in_europe = yes
						surrender_progress < 0.8
					}
					TUR = {
						is_in_faction_with = ROOT
						is_major = yes
						is_subject = no
						surrender_progress < 0.8
					}
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			if = {
				limit = { is_ai = no }
				every_country = {
					limit = {
						is_ally_with = ROOT
						is_major = yes
						has_capitulated = no
					}
					add_timed_idea = {
						idea = has_scripted_peace
						days = 20
					}
				}
			}
			if = { #Does Germany flee to MAF?
				limit = {
					MAF = {
						is_subject_of = GER
						965 = { is_controlled_by = MAF } #Dar-es-Salaam
						surrender_progress < 0.30
						OR = {
							is_ai = yes
							GER = { is_ai = yes }
						}
					}
				}
				set_global_flag = GER_flees_to_africa
			}
			news_event = worldnews.145

			LEC = {
				country_event = { id = legation_cities_events.100 days = 1 }
			}

			if = {
				limit = {
					DEN = {
						has_war = no
						has_soclib_or_socdem_government = yes
					}
				}
				DEN = {
					country_event = { id = denmark_events.121 days = 30 }
				}
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event capitulation_events.1 option capitulation_events.1.a"
		name = capitulation_events.1.a #Assume control of Mittelafrika. We shall fight on from there
		trigger = {
			has_global_flag = GER_flees_to_africa
		}
		set_country_flag = GER_capitulated
		if = {
			limit = { MAF = { is_ai = no } }
			MAF = { country_event = capitulation_events.2 }
		}
		else = {
			MAF = { change_tag_from = ROOT }
			hidden_effect = { germany_flees_to_africa = yes }
		}
	}
	option = {
		log = "[GetLogInfo]: event capitulation_events.1 option capitulation_events.1.b"
		name = capitulation_events.1.b #There is nowhere to run. Surrender!
		trigger = {
			NOT = { has_global_flag = GER_flees_to_africa }
		}
		set_country_flag = GER_capitulated
		custom_effect_tooltip = GER_will_capitulate
		hidden_effect = { germany_capitulates = yes }
	}
}
