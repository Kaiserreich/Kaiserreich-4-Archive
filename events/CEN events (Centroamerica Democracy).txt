		log = "[GetLogInfo]: event cendem.201 option cendem.201.a"
		name = cendem.201.a
		ai_chance = {
			base = 100
			modifier = {
				factor = 0
				has_any_authoritarian_government = yes

				factor = 0

			}
		}

			add_opinion_modifier = {
				target = ROOT
				modifier = KR_pleased
			}

				limit = { is_claimed_by = ROOT }
				remove_claim_by = ROOT
			}

		}
	}

	option = { #I'm a bad player who enjoys the suffering of Central America
		log = "[GetLogInfo]: event cendem.201 option cendem.201.b"
		name = cendem.201.b
		ai_chance = {
			base = 0
			modifier = {
				add = 1
				has_any_authoritarian_government = yes
				FROM.FROM = { any_owned_state = { is_claimed_by = ROOT } }
			}
			modifier = {
				add = 1
				has_government = totalist
				FROM.FROM = { has_socialist_government = no }
			}
		}
		add_political_power = -50
		FROM.FROM = {
			add_opinion_modifier = {
				target = ROOT

			}
			country_event = { id = cendem.203 days = 2 }
