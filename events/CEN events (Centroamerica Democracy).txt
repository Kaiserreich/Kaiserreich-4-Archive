		}

				character = HON_jose_maria_moncada_tapia

			}
		}
		else = {
			add_country_leader_role = {
				character = HON_jose_maria_moncada_tapia

					desc = HON_jose_ramon_adolfo_villeda_morales_leader_desc

			}
		}
	}
	after = {
		hidden_effect = {

		}
	}
