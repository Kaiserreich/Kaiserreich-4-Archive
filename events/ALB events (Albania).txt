### Albanian-Bulgarian friendship
country_event = {
	id = albania_events.13
	title = albania_events.13.t
	desc = albania_events.13.d
	picture = GFX_report_event_generic_sign_treaty2

	is_triggered_only = yes

	option = { #Yes
		log = "[GetLogInfo]: event albania_events.13 option albania_events.13.a"
		name = albania_events.13.a
		ai_chance = { base = 50 }
		FROM = {
			country_event = albania_events.14
			if = {
				limit = { ROOT = { has_socialist_government = yes } }
				custom_effect_tooltip = tooltip_white_line
				effect_tooltip = { ALB_switch_to_radsoc = yes }
			}
		}
		set_country_flag = ALB_aligned
		custom_effect_tooltip = tooltip_white_line
		effect_tooltip = {
			if = {
				limit = { is_in_faction = yes }
				faction_leader = { add_to_faction = FROM }
			}
			else = {
				custom_effect_tooltip = ALB_will_join_BUL_factions
				diplomatic_relation = {
					country = FROM
					relation = non_aggression_pact
				}
				give_military_access = FROM
				give_guarantee = FROM
				FROM = {
					give_military_access = ROOT
					give_guarantee = ROOT
				}
			}
		}
	}

	option = { #Nah m8
		log = "[GetLogInfo]: event albania_events.13 option albania_events.13.b"
		name = albania_events.13.b
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				FROM = {
					OR = {
						has_idea = ALB_trade_with_bulgaria_idea
						has_country_flag = SER_supported_bulgaria
						has_country_flag = ALB_joined_BUL_war
						is_ai = no
					}
				}
			}
			modifier = {
				factor = 0
				OR = {
					has_game_rule = { rule = ALB_forpol option = ALB_FORPOL_BUL }
					has_game_rule = { rule = ALB_forpol option = ALB_FORPOL_4BW }
				}
			}
		}
		FROM = { country_event = albania_events.15 }
	}
}

### Bulgaria accepts Albanian alliance
country_event = {
	id = albania_events.14
	title = albania_events.14.t
	desc = albania_events.14.d
	picture = GFX_report_event_generic_read_write

	trigger = {
		is_subject = no
		country_exists = FROM
		NOT = { has_war_with = FROM }
	}

	is_triggered_only = yes

	option = { #Toast to our Bulgarian friends!
		log = "[GetLogInfo]: event albania_events.14 option albania_events.14.a"
		name = albania_events.14.a
		add_political_power = 45
		add_opinion_modifier = {
			target = FROM
			modifier = KR_close_allies
		}
		reverse_add_opinion_modifier = {
			target = FROM
			modifier = KR_close_allies
		}
		if = {
			limit = { FROM = { has_socialist_government = yes } }
			custom_effect_tooltip = tooltip_white_line
			ALB_switch_to_radsoc = yes
			custom_effect_tooltip = tooltip_white_line
		}
		if = {
			limit = { FROM = { is_in_faction = yes } }
			var:FROM.faction_leader = { add_to_faction = ROOT }
		}
		else = {
			custom_effect_tooltip = ALB_will_join_BUL_factions
			diplomatic_relation = {
				country = FROM
				relation = non_aggression_pact
			}
			give_military_access = FROM
			give_guarantee = FROM
			FROM = {
				give_military_access = ROOT
				give_guarantee = ROOT
			}
		}
	}
}
