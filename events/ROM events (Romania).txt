		name = romania_events.46.a

		if = {
			limit = { NOT = { has_war_with = event_target:ROM_transylvania_target } }

				target = event_target:ROM_transylvania_target
				type = take_state_focus
				generator = { 789 84 76 82 83 }
			}
		}
		effect_tooltip = {
			if = {
				limit = { has_war_with = TRS }
				annex_country = { target = TRS transfer_troops = yes }
			}
			TRS = {
				every_core_state = {
					limit = { owner = { is_ally_with = event_target:ROM_transylvania_target } }
					set_state_owner_to = ROOT
					add_dynamic_modifier = { modifier = ROM_transylvanian_collaboration }
				}
			}
		}
		hidden_effect = {
			every_other_country = {
				limit = { is_ally_with = event_target:ROM_transylvania_target }
				country_event = romania_events.50
			}

			TRS = {
				#create array for valid targets to teleport
				every_core_state = {
					every_neighbor_state = {
						limit = {
							NOT = { is_core_of = TRS }
							is_controlled_by = owner
							owner = { is_ally_with = event_target:ROM_transylvania_target }
						}
						add_to_temp_array = { teleport_targets = THIS }
					}
				}

				every_core_state = {
					limit = {
						owner = {
							OR = {
								tag = ROOT
								has_war_with = ROOT
								is_ally_with = event_target:ROM_transylvania_target
							}
						}
						NOT = { state = 875 }
					}

					#Remove AUS/HUN armies
					if = {
						limit = { has_variable = teleport_targets^num }
						teleport_armies = {
							limit = {
								OR = {
									has_war_with = ROOT
									is_ally_with = event_target:ROM_transylvania_target
								}
							}
							to_state_array = teleport_targets
						}
					}
					else = {
						teleport_armies = {
							limit = {
								OR = {
									has_war_with = ROOT
									is_ally_with = event_target:ROM_transylvania_target
								}
							}
							to_state_array = owned_controlled_states
						}
					}

					#remove any AUS/HUN units that might have crossed over into Romania itself
					every_neighbor_state = {
						limit = {
							event_target:ROM_transylvania_target = {
								OR = {
									divisions_in_state = { state = PREV size > 0 }
									if = {
										limit = { is_in_faction = yes }
										any_allied_country = { divisions_in_state = { state = PREV.PREV size > 0 } }
									}
									else = {
										any_subject_country = { divisions_in_state = { state = PREV.PREV size > 0 } }
									}
								}
							}
							owner = { has_war_with = event_target:ROM_transylvania_target }
							OR = {
								region = 15 #Romania
								region = 24 #Central Balkans
								region = 144 #Transylvania
