### Ask for the Iron Crown of Lombardy
country_event = {
	id = lombardy_events.1
	title = lombardy_events.1.t
	picture = GFX_report_event_ITA_milan

	desc = {
		text = lombardy_events.1.d.aus
		trigger = { country_exists = AUS }
	}
	desc = {
		text = lombardy_events.1.d.sic
		trigger = {
			NOT = { country_exists = AUS }
			country_exists = SIC
		}
	}
	desc = {
		text = lombardy_events.1.d.vienna
		trigger = {
			NOT = { country_exists = AUS }
			NOT = { country_exists = SIC }
		}
	}

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event lombardy_events.1 option lombardy_events.1.a"
		name = lombardy_events.1.a
		trigger = { country_exists = AUS }
		AUS = {
			country_event = lombardy_events.2
		}
	}
	option = {
		log = "[GetLogInfo]: event lombardy_events.1 option lombardy_events.1.b"
		name = lombardy_events.1.b
		trigger = {
			NOT = { country_exists = AUS }
			country_exists = SIC
		}
		SIC = {
			country_event = lombardy_events.2
		}
	}
	option = {
		log = "[GetLogInfo]: event lombardy_events.1 option lombardy_events.1.c"
		name = lombardy_events.1.c
		trigger = {
			NOT = { country_exists = AUS }
			NOT = { country_exists = SIC }
		}
		random_country = {
			limit = { owns_state = 4 }
			country_event = lombardy_events.2
		}
	}
}

### Lombardy asks for the Iron Crown
country_event = {
	id = lombardy_events.2
	title = lombardy_events.2.t
	desc = lombardy_events.2.d
	picture = GFX_report_event_ITA_milan

	is_triggered_only = yes

	#Return it
	option = {
		log = "[GetLogInfo]: event lombardy_events.2 option lombardy_events.2.a"
		name = lombardy_events.2.a
		ai_chance = {
			base = 90
			modifier = {
				factor = 0
				has_socialist_government = yes
			}
		}
		add_political_power = -10
		FROM = {
			country_event = { id = lombardy_events.3 days = 3 }
		}
		LOM = {
			add_opinion_modifier = {
				target = ROOT
				modifier = LOM_returned_crown
			}
		}
	}

	#Refuse!
	option = {
		log = "[GetLogInfo]: event lombardy_events.2 option lombardy_events.2.b"
		name = lombardy_events.2.b
		trigger = {
			LOM = {
				is_ally_of_ROOT = no
			}
		}
		ai_chance = { base = 10 }
		FROM = {
			country_event = { id = lombardy_events.4 days = 3 }
		}
		LOM = {
			add_opinion_modifier = {
				target = ROOT
				modifier = LOM_refused_crown
			}
		}
	}
}
