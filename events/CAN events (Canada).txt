# Death of King George V
country_event = {
	id = canada_events.4
	title = canada_events.4.t
	desc = canada_events.4.d
	picture = GFX_report_event_CAN_death_george

	is_triggered_only = yes

	trigger = {
		has_country_leader = { character = GBR_george_frederick_windsor }
	}

	option = {
		log = "[GetLogInfo]: event canada_events.4 option canada_events.4.a"
		name = canada_events.4.a
		GBR_george_frederick_windsor = { kill_character = yes }
		set_variable = { king_popularity = 2 }
		set_variable = { global.king_edward_story = 0 }
		add_stability = -0.05
		custom_effect_tooltip = CAN_king_speech_unlocked
		hidden_effect = {
			GBR_create_edward = yes
			# official coronation
			country_event = { id = canada_events.5 days = 90 }
			# country-specific death events
			if = {
				limit = { country_exists = RAJ }
				RAJ = { country_event = dominion_of_india_events.3 }
			}
			if = {
				limit = { country_exists = AST }
				AST = { country_event = australasia_events.3 }
			}
			if = {
				limit = { country_exists = NFA }
				NFA = { country_event = french_republic_events.4 }
			}
			if = {
				limit = { country_exists = SAF }
				SAF = { country_event = south_africa_events.158 }
			}
			if = {
				limit = { country_exists = IRE }
				IRE = { country_event = ireland_events.2 }
			}
			# main news events
			news_event = { id = canada_flavour_events.5 days = 3 }
		}
	}
}

# American refugees flood into state (repeating event)
country_event = {
	id = canada_events.202
	title = canada_events.202.t
	desc = canada_events.202.d
	picture = GFX_report_event_CAN_refugees

	trigger = {
		tag = CAN
		is_subject = no
		has_global_flag = USA_civil_war_begun_flag
		check_variable = { refugee_states < 10 }
		NOT = {
			has_war_with = USA
			has_war_with = WCA
			has_war_with = ACC
			has_war_with = APG
			has_war_with = NEE
			has_idea = CAN_refugee_crisis_sec
			has_country_flag = CAN_ref_shipped_to_nee
			has_global_flag = USA_civil_war_over_flag
		}
		any_owned_state = {
			is_controlled_by = CAN
			NOT = { has_dynamic_modifier = { modifier = CAN_american_refugee_modifier } }
			any_neighbor_state = {
				state_population_k > 750
				OR = {
					is_owned_by = WCA
					controller = {
						is_american_tag = yes
						NOT = { owns_state = PREV }
					}
				}
			}
		}
	}

	mean_time_to_happen = {
		days = 60
		modifier = {
			factor = 0.5
			check_variable = { refugee_states < 1 }
		}
		modifier = {
			factor = 1.5
			check_variable = { refugee_states > 2 }
		}
	}

	immediate = {
		hidden_effect = {
			random_owned_controlled_state = {
				limit = {
					NOT = { has_dynamic_modifier = { modifier = CAN_american_refugee_modifier } }
					any_neighbor_state = {
						state_population_k > 750
						OR = {
							is_owned_by = WCA
							controller = {
								is_american_tag = yes
								NOT = { owns_state = PREV }
							}
						}
					}
				}
				save_event_target_as = refugee_state
				set_variable = { CAN_american_refugee_local_supplies = -0.2 }
				random_neighbor_state = {
					limit = {
						state_population_k > 750
						OR = {
							is_owned_by = WCA
							controller = {
								is_american_tag = yes
								NOT = { owns_state = PREV }
							}
						}
					}
					save_event_target_as = flee_from_state
				}
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event canada_events.202 option canada_events.202.a"
		name = canada_events.202.a
		event_target:refugee_state = {
			custom_effect_tooltip = tooltip_line_break
			add_dynamic_modifier = { modifier = CAN_american_refugee_modifier }
		}
		if = {
			limit = { has_idea = CAN_refugee_crisis_help }
			add_manpower = 7500
		}
		if = {
			limit = { has_idea = CAN_refugee_crisis_sam }
			add_manpower = 12500
		}
		hidden_effect = {
			add_to_variable = { refugee_states = 1 }
		}
	}
}
