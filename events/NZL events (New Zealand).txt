# Canada considers accepting New Zealand into the Entente
country_event = {
	id = nzl.15
	immediate = { log = "[GetDateText]: [Root.GetName]: event nzl.15 FROM = [From.GetName]" }
	title = nzl.15.t
	desc = nzl.15.d
	picture = GFX_report_event_ast_navy

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		is_faction_leader = yes
		has_socialist_government = no
		NOT = {
			is_in_faction_with = AST
			has_war_with = AST
			has_war_with = NZL
			has_capitulated = yes
		}
		NZL = {
			has_war_with = AST
			is_valid_peace_target = yes
			has_socialist_government = no
			has_capitulated = no
		}
	}

	option = {
		name = nzl.15.a # Yes, of course we'll help them!
		ai_chance = {
			base = 75
			modifier = {
				factor = 1.25
				has_war = no
			}
		}
		add_political_power = 50
		add_to_faction = NZL
		add_to_war = {
			targeted_alliance = NZL
			enemy = AST
		}
	}
	option = {
		name = nzl.15.b # No, we cannot afford to become embroiled in another war
		ai_chance = {
			base = 25
			modifier = {
				factor = 3
				surrender_progress > 0.20
			}
			modifier = {
				factor = 2
				OR = {
					has_war_with = ENG
					has_war_with = GER
				}
			}
			modifier = {
				factor = 0
				has_country_flag = AST_supporting_new_guard_coup
			}
			modifier = {
				factor = 0
				has_game_rule = { rule = NZL_succession option = NZL_SUCCESSION_ENT }
			}
		}
		if = {
			limit = { has_country_flag = AST_supporting_new_guard_coup }
			add_political_power = -200
		}
		else = {
			add_political_power = -50
		}
		NZL = {
			country_event = nzl.16
		}
	}
}
