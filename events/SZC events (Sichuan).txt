		log = "[GetLogInfo]: event sichuan_events.1005 option 1"

			limit = {
				country_exists = QIE

					china_is_aligned_with_zhili = yes
					NOT = { has_country_flag = SZC_refused_armament }
				}
			}
			QIE = { country_event = sichuan_events.16 }
		}
		else_if = {
			limit = {
				country_exists = ANQ

					china_is_aligned_with_zhili = yes
					NOT = { has_country_flag = SZC_refused_armament }
				}
			}
			ANQ = { country_event = sichuan_events.16 }
		}
		else_if = {
			limit = {
				country_exists = LEP
				LEP = {
					NOT = { has_country_flag = SZC_refused_armament }
				}
			}
			LEP = { country_event = sichuan_events.16 }
		}
	}
