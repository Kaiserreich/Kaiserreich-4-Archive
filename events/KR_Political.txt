country_event = {
	id = kr.political.1
	title = kr.political.reduction
	picture = GFX_report_event_generic_rally_3

	desc = {
		text = kr.political.vgtog
		trigger = { is_at_good_stability = yes }
	}
	desc = {
		text = kr.political.vgtonorm
		trigger = { is_at_normal_stability = yes }
	}
	desc = {
		text = kr.political.vgtolow
		trigger = { is_at_low_stability = yes }
	}
	desc = {
		text = kr.political.vgtovlow
		trigger = { is_at_very_low_stability = yes }
	}
	desc = {
		text = kr.political.vgtodis
		trigger = { is_at_disaster_stability = yes }
	}

	trigger = {
		has_idea = very_good_stability
		NOT = { is_at_very_good_stability = yes }
	}

	mean_time_to_happen = {
		days = 3
	}

	option = {
		name = kr.political.reduction.a
		if = {
			limit = { is_at_good_stability = yes }
			effect_tooltip = {
				swap_ideas = {
					remove_idea = very_good_stability
					add_idea = good_stability
				}
			}
		}
		else_if = {
			limit = { is_at_normal_stability = yes }
			effect_tooltip = {
				swap_ideas = {
					remove_idea = very_good_stability
					add_idea = normal_stability
				}
			}
		}
		else_if = {
			limit = { is_at_low_stability = yes }
			effect_tooltip = {
				swap_ideas = {
					remove_idea = very_good_stability
					add_idea = low_stability
				}
			}
		}
		else_if = {
			limit = { is_at_very_low_stability = yes }
			effect_tooltip = {
				swap_ideas = {
					remove_idea = very_good_stability
					add_idea = very_low_stability
				}
			}
		}
		else_if = {
			limit = { is_at_disaster_stability = yes }
			effect_tooltip = {
				swap_ideas = {
					remove_idea = very_good_stability
					add_idea = disaster
				}
			}
		}
		hidden_effect = {
			if = {
				limit = { has_stability < 0.86 }
				swap_ideas = {
					remove_idea = very_good_stability
					add_idea = good_stability
				}
			}
			if = {
				limit = { has_stability < 0.66 }
				swap_ideas = {
					remove_idea = good_stability
					add_idea = normal_stability
				}
			}
			if = {
				limit = { has_stability < 0.46 }
				swap_ideas = {
					remove_idea = normal_stability
					add_idea = low_stability
				}
			}
			if = {
				limit = { has_stability < 0.26 }
				swap_ideas = {
					remove_idea = low_stability
					add_idea = very_low_stability
				}
			}
			if = {
				limit = { has_stability < 0.01 }
				swap_ideas = {
					remove_idea = very_low_stability
					add_idea = disaster
				}
			}
		}
	}
}

### very_high_popularity - Very High Popularity
country_event = {
	id = kr.political.11
	title = kr.political.11.t
	desc = kr.political.11.d
	picture = popularity_rising

	fire_only_once = no

	trigger = {
		NOT = {
			has_idea = very_high_popularity
			has_country_flag = unpopular_government
		}
		OR = {
			AND = {
				has_government = totalist
				totalist > 0.799
			}
			AND = {
				has_government = radical_socialist
				radical_socialist > 0.799
			}
			AND = {
				has_government = syndicalist
				syndicalist > 0.799
			}
			AND = {
				has_government = social_democrat
				social_democrat > 0.799
			}
			AND = {
				has_government = social_liberal
				social_liberal > 0.799
			}
			AND = {
				has_government = market_liberal
				market_liberal > 0.799
			}
			AND = {
				has_government = social_conservative
				social_conservative > 0.799
			}
			AND = {
				has_government = authoritarian_democrat
				authoritarian_democrat > 0.799
			}
			AND = {
				has_government = paternal_autocrat
				paternal_autocrat > 0.799
			}
			AND = {
				has_government = national_populist
				national_populist > 0.799
			}
		}
	}

	mean_time_to_happen = {
		days = 2
	}

	option = {
		name = kr.political.11.a
		if = {
			limit = { has_idea = very_low_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = very_low_popularity add_idea = very_high_popularity } }
			hidden_effect = { swap_ideas = { remove_idea = very_low_popularity add_idea = low_popularity } }
		}
		if = {
			limit = { has_idea = low_popularity }
			effect_tooltip = { swap_ideas = { remove_idea = low_popularity add_idea = very_high_popularity } }
			hidden_effect = { swap_ideas = { remove_idea = low_popularity add_idea = normal_popularity } }
		}
