#rename city
country_event = {
	id = germany_puppet_events.1
	title = germany_puppet_events.1.t
	desc = germany_puppet_events.1.d
	picture = GFX_report_event_generic_big_city

	is_triggered_only = yes

	trigger = {
		check_variable = { 67.zabrze_german_name = 0 }
		has_socialist_government = yes
		owns_state = 67
		controls_province = 11467
	}

	option = { #Marx
		log = "[GetLogInfo]: event germany_puppet_events.1 option germany_puppet_events.1.a"
		name = germany_puppet_events.1.a
		ai_chance = { base = 60 }
		set_variable = { 67.zabrze_german_name = 1 }
		effect_tooltip = {
			set_province_name = { id = 11467 name = VICTORY_POINTS_11467_KARL_MARX_STADT }
		}
	}

	option = { #nothing political
		log = "[GetLogInfo]: event germany_puppet_events.1 option germany_puppet_events.1.b"
		name = germany_puppet_events.1.b
		ai_chance = { base = 40 }
		set_variable = { 67.zabrze_german_name = 2 }
		effect_tooltip = {
			set_province_name = { id = 11467 name = VICTORY_POINTS_11467_EISENHUTTENSTADT }
		}
	}

	after = {
		67 = { rename_state_effect = yes }
	}
}

#reset city name if retaken
country_event = {
	id = germany_puppet_events.2
	title = germany_puppet_events.2.t
	desc = germany_puppet_events.2.d
	picture = GFX_report_event_generic_big_city

	is_triggered_only = yes

	trigger = {
		NOT = { check_variable = { 67.zabrze_german_name = 0 } }
		has_socialist_government = no
		owns_state = 67
		controls_province = 11467
	}

	option = { #restore it
		log = "[GetLogInfo]: event germany_puppet_events.2 option germany_puppet_events.2.a"
		name = germany_puppet_events.2.a
		clear_variable = 67.zabrze_german_name
		effect_tooltip = {
			set_province_name = { id = 11467 name = VICTORY_POINTS_11467 }
		}
		67 = { rename_state_effect = yes }
	}
}

country_event = {
	id = germany_puppet_events.3
	title = germany_puppet_events.3.t
	desc = germany_puppet_events.3.d
	picture = GFX_report_event_generic_big_city

	is_triggered_only = yes

	option = { #Grant independence
		log = "[GetLogInfo]: event germany_puppet_events.3 option germany_puppet_events.3.a"
		name = germany_puppet_events.3.a
		ai_chance = { base = 100 }
		set_autonomy = {
			target = FROM
			autonomy_state = autonomy_free
		}
		if = {
			limit = { has_subject = RHI }
			FROM = {
				annex_country = {
					target = RHI
					transfer_troops = yes
				}
			}
		}
		custom_effect_tooltip = GRU_independence_tt
	}

	option = { #Maintian Control
		log = "[GetLogInfo]: event germany_puppet_events.3 option germany_puppet_events.3.b"
		name = germany_puppet_events.3.b
		ai_chance = { base = 0 }
		add_political_power = -50
		FROM = {
			add_popularity = {
				ideology = national_populist
				popularity = 0.15
			}
		}
	}
}
