	id = KRBAT.9
	hidden = yes

	trigger = {
		has_idea = BAT_dualism
	}

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event KRBAT.9 option 1"
		if = {
			limit = {
				check_variable = { ROOT.political_power_growth < 0 }
			}
			set_temp_variable = { BAT_pp_delta = BAT.political_power_growth }
			multiply_temp_variable = { BAT_pp_delta = -1 }
			add_political_power = BAT_pp_delta
		}
		country_event = { id = KRBAT.9 days = 1 }
	}
}
