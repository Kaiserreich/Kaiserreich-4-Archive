country_event = { # Ungern-Sternberg's Gambit
	id = united_baltic_duchy_events.8
	title = united_baltic_duchy_events.8.t
	desc = united_baltic_duchy_events.8.d
	picture = GFX_report_event_Baltic_Landtag

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { is_ai = yes }
			if = {
				limit = { has_game_rule = { rule = BAT_collapse option = BAT_COLLAPSE_YES } }
				set_country_flag = BAT_AI_collapse
			}
			else_if = {
				limit = { has_game_rule = { rule = BAT_path option = BAT_PATH_SOCLIB } }
				set_country_flag = BAT_AI_liberal
			}
			else_if = {
				limit = { has_game_rule = { rule = BAT_path option = BAT_PATH_PATAUT } }
				set_country_flag = BAT_AI_knighthoods
			}
			else_if = {
				limit = { has_game_rule = { rule = BAT_path option = BAT_PATH_NATPOP } }
				set_country_flag = BAT_AI_brotherhood
			}
			else = {
				random_list = {
					25 = { set_country_flag = BAT_AI_liberal }
					25 = { set_country_flag = BAT_AI_knighthoods }
					25 = { set_country_flag = BAT_AI_brotherhood }
					25 = { set_country_flag = BAT_AI_collapse }
				}
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event united_baltic_duchy_events.8 option united_baltic_duchy_events.8.a"
		name = united_baltic_duchy_events.8.a # Oh scheisse (Dualism Begins)
	}
	option = {
		log = "[GetLogInfo]: event united_baltic_duchy_events.8 option united_baltic_duchy_events.8.b"
		name = united_baltic_duchy_events.8.b # Who is von Ungern-Sternberg? (Dualism Begins)
		ai_chance = { base = 0 }
		country_event = united_baltic_duchy_events.47
	}
	after = {
		set_political_power = 0
		add_stability = -0.05
		add_ideas = BAT_dualism
		unlock_decision_category_tooltip = BAT_dualism_decision_category
		hidden_effect = {
			activate_mission = BAT_duke_timer
			country_event = { id = united_baltic_duchy_events.9 days = 1 }
			country_event = { id = united_baltic_duchy_events.68 days = 7 }
			country_event = { id = united_baltic_duchy_events.111 days = 30 }
		}
	}
