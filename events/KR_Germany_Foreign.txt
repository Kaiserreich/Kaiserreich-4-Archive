country_event = { #The Kaiserreich Besieged. Two-front war with France and Russia, only available when Russia attacks
	id = gerfor.187
	immediate = {log = "[GetDateText]: [Root.GetName]: event gerfor.187"}
	title = gerfor.187.t
	desc = gerfor.187.d
	picture = towar

	fire_only_once = yes

	trigger = { #This is to check and make sure that it's a defensive war with Russia.
		tag = GER
		has_war_with = FRA
		OR = {
			has_war_with = SOV
			has_war_with = RUS
		}
		NOT = {
			has_offensive_war_with = SOV
		}
		NOT = {
			has_offensive_war_with = RUS
		}
	}

	option = { # Activates defense and org buff for 730 days.
		name = gerfor.187.a
		add_timed_idea = {
			idea = GER_two_front_idea
			days = 730
		}
	}
}
