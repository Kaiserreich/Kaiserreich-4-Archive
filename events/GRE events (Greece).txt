		}
	}

	option = {
		log = "[GetLogInfo]: event greece_events.109 option greece_events.109.a"
		name = greece_events.109.a

			}
			custom_effect_tooltip = GRE_referendum_is_rigged

					has_country_flag = GRE_monarchy_rig

					}

		}

			limit = {
				NOT = {
					has_country_flag = GRE_monarchy_rig
				}
