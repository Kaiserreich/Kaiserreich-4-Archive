country_event = { #Fate of Central Asia
	id = annex.3000
	immediate = { log = "[GetDateText]: [Root.GetName]: event annex.3000 FROM = [From.GetName]" }
	title = annexation_CA_CONQ
	desc = annexation_CA_CONQ_desc
	picture = GFX_report_event_SIK_scenery
	is_triggered_only = yes

	immediate = {
		hidden_effect = {
			add_to_temp_array = { core_removal_array = ALO }
			add_to_temp_array = { core_removal_array = TRK }
			remove_all_cores_and_claims_of_tags = yes
			for_each_scope_loop = {
				array = RUS.RUS_kazakhstan_region
				add_core_of = ALO
			}
			for_each_scope_loop = {
				array = RUS.RUS_turkestan_region
				add_core_of = TRK
			}
		}
	}

	option = { #CAF
		name = annex.3000.a
		ai_chance = {
			base = 20
			modifier = {
				factor = 0
				is_russian_tag = yes
			}
		}
		set_temp_variable = { tag_to_release = ALO }
		add_to_temp_array = { target_cores_to_release = ALO }
		add_to_temp_array = { target_cores_to_release = TRK }
		release_targeted_tag_from_core_regions_and_add_cores = yes
		var:tag_to_release = {
			set_cosmetic_tag = ALO_CAF
		}
	}

	option = { #Divide major
		name = annex.3000.b
		ai_chance = {
			base = 40
			modifier = {
				factor = 0
				is_russian_tag = yes
			}
		}
		#Alash
		set_temp_variable = { tag_to_release = ALO }
		release_targeted_tag = yes
		if = {
			limit = { is_russian_tag = no }
			ALO = { set_cosmetic_tag = ALO_KAZ }
		}

		#Turkestan
		set_temp_variable = { tag_to_release = TRK }
		release_targeted_tag = yes
	}

	option = { #divide minors
		name = annex.3000.c
		ai_chance = {
			base = 35
			modifier = {
				factor = 0
				is_russian_tag = yes
			}
		}
		#Kyrgyzstan
		hidden_effect = {
			586 = { #Kokand
				add_core_of = KYR
				remove_core_of = ALO
			}
		}
		set_temp_variable = { tag_to_release = KYR }
		release_targeted_tag = yes

		#Tajikistan
		hidden_effect = {
			804 = { #Pamir
				add_core_of = TAJ
				remove_core_of = TRK
			}
			1042 = { #Bandakhshan
				add_core_of = TAJ
				remove_core_of = TRK
			}
		}
		set_temp_variable = { tag_to_release = TAJ }
		release_targeted_tag = yes

		#Uzbekistan
		hidden_effect = {
			405 = { #Bukhara
				add_core_of = UZB
				remove_core_of = TRK
			}
			407 = { #Tashkent
				add_core_of = UZB
				remove_core_of = TRK
			}
			712 = { #Kyzylorda
				add_core_of = UZB
				remove_core_of = TRK
			}
			732 = { #Samarkand
				add_core_of = UZB
				remove_core_of = TRK
			}
			861 = { #Khorezm
				add_core_of = UZB
				remove_core_of = TRK
			}
		}
		set_temp_variable = { tag_to_release = UZB }
		release_targeted_tag = yes

		#Alash
		hidden_effect = {
			712 = { #Kyzylorda
				add_core_of = ALO
				remove_core_of = TRK
			}
		}
		set_temp_variable = { tag_to_release = ALO }
		release_targeted_tag = yes
		if = {
			limit = { is_russian_tag = no }
			ALO = { set_cosmetic_tag = ALO_KAZ }
		}

		#Turkestan
		set_temp_variable = { tag_to_release = TRK }
		release_targeted_tag = yes
	}

	option = { #Integrate
		name = annex.3000.d
		trigger = {
			is_russian_tag = yes
		}
		ai_chance = {
			base = 100
		}
		every_state = {
			limit = {
				region = 136 #Central Asia
				is_claim_or_core_of_ROOT = no
			}
			add_claim_by = ROOT
		}
		add_political_power = -80
		add_to_array = { RUS_integration_array = ALO }
		add_to_array = { RUS_integration_array = TRK }
		recheck_annexations = yes
	}

	option = { #give to Russian ally
		name = annex.3000.e
		trigger = {
			is_russian_tag = no
			var:572.owner = {
				is_russian_tag = yes
				is_ally_with = ROOT
			}
		}
		ai_chance = {
			base = 80
			modifier = {
				factor = 0
				is_german_tag = yes
			}
		}
		add_to_temp_array = { target_cores_to_release = ALO }
		add_to_temp_array = { target_cores_to_release = TRK }
		set_temp_variable = { tag_to_release = 572.owner }
		transfer_territory_from_core_regions = yes
	}

	option = { #occupy
		name = annex.occupation
		trigger = {
			is_ai = no
			is_russian_tag = no
		}
		set_country_flag = CA_CONQ
		occupy_huge_country = yes
		hidden_effect = {
			for_each_scope_loop = {
				array = RUS.RUS_turkestan_region
				add_core_of = TRK
			}
		}
	}
}
