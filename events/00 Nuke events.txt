# Hidden event to set flag on nuke dropped
state_event = {
	id = nuke_dropped.0

	trigger = {
		wants_world_news = yes #Global news event
		NOT = { has_state_flag = state_was_nuked_once }
	}

	is_triggered_only = yes
	hidden = yes

	immediate = {
		set_state_flag = state_was_nuked_once
	}

	option = {
		if = {
			limit = { state = 529 } #San'yo
			ROOT = { news_event = { id = nuke_dropped.2 hours = 1 } }
		}
		else_if = {
			limit = { state = 528 } #North Kyushu
			ROOT = { news_event = { id = nuke_dropped.3 hours = 1 } }
		}
		else_if = {
			limit = { state = 64 } #Brandenburg
			ROOT = { news_event = { id = nuke_dropped.4 hours = 1 } }
		}
		else_if = {
			limit = { state = 126 } #Greater London
			ROOT = { news_event = { id = nuke_dropped.5 hours = 1 } }
		}
		else_if = {
			limit = { state = 531 } #Kansai
			ROOT = { news_event = { id = nuke_dropped.6 hours = 1 } }
		}
		else_if = {
			limit = { state = 282 } #Tokyo
			ROOT = { news_event = { id = nuke_dropped.7 hours = 1 } }
		}
		else_if = {
			limit = { state = 361 } #Maryland
			ROOT = { news_event = { id = nuke_dropped.8 hours = 1 } }
		}
		else_if = {
			limit = { state = 2 } #Latium
			ROOT = { news_event = { id = nuke_dropped.9 hours = 1 } }
		}
		else_if = {
			limit = { state = 16 } #Ile-de-France

		else_if = {
			limit = { state = 219 } #Moscow

		else_if = {
			limit = { state = 37 } #Zealand

		}
		else_if = {
			limit = { state = 141 } #Svealand

		}
		else_if = {
			limit = { state = 289 } #Siam

		}
		else = {
			ROOT = { news_event = { id = nuke_dropped.1 hours = 1 } }
		}
	}
