		text = beng.16.d.promotion
		trigger = {
			has_idea = HND_elite_promotion
		}
	}

	fire_only_once = yes

	trigger = {
		tag = HND
		has_completed_focus = HND_indian_elites
		has_idea = HND_backwards_economy_1
		OR = {
			has_idea = HND_elite_resistance
			has_idea = HND_elite_incentive
			has_idea = HND_elite_promotion
		}
	}

	mean_time_to_happen = {
		months = 6
		modifier = {
			factor = 2
			has_idea = HND_elite_resistance
		}
		modifier = {
			factor = 0.5
			has_idea = HND_elite_promotion
		}
		modifier = {
			factor = 0.5
			has_war = yes
		}
	}

	option = {
		log = "[GetLogInfo]: event beng.16 option beng.16.a"
		name = beng.16.a
		if = {
			limit = {
				has_idea = HND_elite_resistance
			}
			remove_ideas = HND_elite_resistance
		}
		if = {
			limit = {
				has_idea = HND_elite_incentive
			}
			remove_ideas = HND_elite_incentive
		}
		if = {
			limit = {
				has_idea = HND_elite_promotion
			}
			remove_ideas = HND_elite_promotion
		}
		swap_ideas = {
			remove_idea = HND_backwards_economy_1
			add_idea = HND_backwards_economy_2
		}
	}
}
