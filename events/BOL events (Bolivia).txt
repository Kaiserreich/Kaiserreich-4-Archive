country_event = { #Bolivia offers to recreate the Peru-Bolivian confederation

	id = bol.17
	title = bol.17.t
	desc = bol.17.d
	picture = GFX_report_event_military_planning

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event bol.17 option bol.17.a"
		name = bol.17.a
		ai_chance = { base = 100 }
		effect_tooltip = {
			every_state = {
				limit = {
					is_core_of = BOL
					is_owned_by = BOL
					is_claim_or_core_of_ROOT = no
				}
				add_core_of = ROOT
			}
			every_state = {
				limit = {
					is_core_of = BOL
					NOT = { is_owned_by = BOL }
					is_claim_or_core_of_ROOT = no
				}
				add_claim_by = ROOT
			}
		}
		BOL = {
			every_enemy_country = {
				ROOT = {
					add_to_war = {
						targeted_alliance = BOL
						enemy = PREV
					}
				}
			}
		}
		hidden_effect = {
			every_state = {
				limit = {
					OR = {
						is_claimed_by = BOL
						is_core_of = BOL
					}
				}
				if = {
					limit = { is_core_of = BOL }
					add_core_of = ROOT
				}
				else = {
					add_claim_by = ROOT
				}
			}
		}
		if = {
			limit = { BOL = { is_ai = no } }
			ROOT = { change_tag_from = BOL }
		}
		annex_country = {
			target = BOL
			transfer_troops = yes
		}
		set_cosmetic_tag = PBC
		unlock_national_focus = PRU_liberate_bolivia
		unlock_national_focus = PRU_summit
		# gain Bolivia's ambition area
		hidden_effect = {
			news_event = worldnews.164
			BOL = {
				every_unit_leader = {
					set_character_flag = BOL_leader
					set_nationality = PRU
				}
			}
		}
	}
	option = {
		log = "[GetLogInfo]: event bol.17 option bol.17.b"
		ai_chance = { base = 0 }
		name = bol.17.b
	}
}
