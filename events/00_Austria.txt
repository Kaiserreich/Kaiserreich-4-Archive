country_event = {
	id = austria.21
	title = austria.21.t
	desc = austria.21.d
	
	fire_only_once = yes
	
	trigger = {
		OR = {
		tag = AUS
		tag = HUN
		}
		AND = {
		HUN = {
			NOT = {
			or = {
				owns_state = 82
				owns_state = 83
				owns_state = 84
				owns_state = 76
				}
			}
		}
		AUS = {
			NOT = {
			OR = {
				owns_state = 82
				owns_state = 83
				owns_state = 84
				owns_state = 76
				}
			}
		}
		
		}
	}
	
	option = { 
		name = austria.21.a
		every_other_country = {
			limit = {
				or = {
				owns_state = 82
				owns_state = 83
				owns_state = 84
				owns_state = 76
				}
			}
			
			country_event = { id =  austria.22 days = 1 }
		}
		
		
	}
	option = { 
		name = austria.21.b
		add_national_unity = -0.05
		
	}

}
