country_event = { # The Resignation of Tom Mann
	id = uob.6
	title = uob.6.t
	desc = uob.6.d
	picture = GFX_report_event_generic_diplomacy1

	is_triggered_only = yes

	trigger = {
		OR = {
			has_country_flag = {
				flag = ENG_tuc_maximist
				value > 1
			}
			has_country_flag = {
				flag = ENG_tuc_federalist
				value > 1
			}
			has_country_flag = {
				flag = ENG_tuc_autonomist
				value > 1
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event uob.6 option uob.6.a"
		name = uob.6.a
		add_political_power = -50
		promote_character = ENG_central_committee
		hidden_effect = {
			if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_maximist
						value > 1
					}
				}
				country_event = { id = uob.7 days = 20 }
			}
			else_if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_federalist
						value > 1
					}
				}
				country_event = { id = uob.8 days = 20 }
			}
			else_if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_autonomist
						value > 1
					}
				}
				country_event = { id = uob.9 days = 20 }
			}
		}
	}
}
