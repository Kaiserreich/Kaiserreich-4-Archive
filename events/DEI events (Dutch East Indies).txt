country_event = { #Reclaimed the birthright as the exiles
	id = dutch_exile_events.40
	title = dutch_exile_events.40.t
	desc = dutch_exile_events.40.d
	picture = GFX_report_event_HOL_rhine_ships

	fire_only_once = yes

	trigger = {
		tag = DEI
		owns_state = 7
	}

	immediate = {
		hidden_effect = {
			HOL = {
				HOL_create_wilhelmina = yes
				set_country_flag = hol_dont_burn_binnenhof
				if = { #Renew Caribbean claims
					limit = { 309 = { is_claim_or_core_of_target = no } }
					add_state_claim = 309
				}
				if = {
					limit = { 1038 = { is_claim_or_core_of_target = no } }
					add_state_claim = 1038
				}
				if = {
					limit = { 695 = { is_claim_or_core_of_target = no } }
					add_state_claim = 695
				}
				if = {
					limit = { 1077 = { is_claim_or_core_of_target = no } }
					add_state_claim = 1077
				}
				remove_targeted_decision = { target = DEI decision = holex_recruit_informants }
				remove_targeted_decision = { target = DEI decision = holex_lure_resistance_into_trap }
				remove_targeted_decision = { target = DEI decision = holex_direct_razzia }
			}
			DEI = {
				remove_targeted_decision = { target = HOL decision = holex_support_the_resistance_cells }
				remove_targeted_decision = { target = HOL decision = holex_establish_hiding_places }
				remove_targeted_decision = { target = HOL decision = holex_forge_false_identities }
				remove_targeted_decision = { target = HOL decision = holex_raise_resistance_fighters }
				remove_targeted_decision = { target = HOL decision = holex_resistance_attack_civilian_fac }
				remove_targeted_decision = { target = HOL decision = holex_bomb_railway }
				remove_targeted_decision = { target = HOL decision = holex_resistance_attack_arms_fac }

				if = {
					limit = { has_country_flag = DEI_raised_resistance_fighters }
					set_division_template_lock = { division_template = "Militie Verzetsstrijders" is_locked = no }
					clr_country_flag = DEI_raised_resistance_fighters
				}
			}
		}
	}

	option = { #Release INS as promised and play as restored HOL
		log = "[GetLogInfo]: event dutch_exile_events.40 option dutch_exile_events.40.a"
		name = dutch_exile_events.40.a
		ai_chance = { base = 100 }
		trigger = {
			has_country_flag = holex_will_release_indonesia
			NOT = { country_exists = INS }
		}
		hidden_effect = {
			INS = { transfer_technology_without_doctrines = yes }
		}
		HOL_restore_exiled_netherlands = yes
		HOL = {
			HOL_DEI_is_annexed = yes
			release = INS
			change_tag_from = ROOT
		}
	}

	option = { #Release INS as promised and play as released INS
		log = "[GetLogInfo]: event dutch_exile_events.40 option dutch_exile_events.40.b"
		name = dutch_exile_events.40.b
		ai_chance = { base = 100 }
		trigger = {
			has_country_flag = holex_will_release_indonesia
			NOT = { country_exists = INS }
		}
		hidden_effect = {
			INS = { transfer_technology_without_doctrines = yes }
		}
		HOL_restore_exiled_netherlands = yes
		HOL = {
			HOL_DEI_is_annexed = yes
			release = INS
		}
		INS = { change_tag_from = ROOT }
	}

	option = { #Give INS equal status as promised - same as the Batavia Conference with minimum coring
		log = "[GetLogInfo]: event dutch_exile_events.40 option dutch_exile_events.40.c"
		name = dutch_exile_events.40.c
		ai_chance = { base = 100 }
		trigger = {
			has_country_flag = holex_will_equal_status_indonesia
		}
		if = {
			limit = { owns_state = 668 }
			668 = { add_core_of = HOL }

		if = {
			limit = { owns_state = 972 }
			972 = { add_core_of = HOL }

		HOL_restore_exiled_netherlands = yes
		HOL = {
			HOL_DEI_is_annexed = yes
			change_tag_from = ROOT
		}
	}

	option = { #Didn't promise to give the Indonesians anything and still managed to restore the birthright - play as DEI
