	title = hawaii_events.1.t
	desc = hawaii_events.1.d

				add_claim_by = USA

					limit = { owner = { is_american_tag = yes } }

				if = {
					limit = { has_resistance = yes }
					set_temp_variable = { previous_compliance = compliance }

			transfer_ship = { prefer_name = "USS Breese" type = destroyer target = event_target:system }
			transfer_ship = { prefer_name = "USS Ramsay" type = destroyer target = event_target:system }
			transfer_ship = { prefer_name = "USS Tattnall" type = destroyer target = event_target:system }

			transfer_ship = { prefer_name = "USS S15" type = submarine target = event_target:system }
			transfer_ship = { prefer_name = "USS S16" type = submarine target = event_target:system }

			transfer_ship = { prefer_name = "USS S19" type = submarine target = event_target:system }
			transfer_ship = { prefer_name = "USS S20" type = submarine target = event_target:system }
			transfer_ship = { prefer_name = "USS S21" type = submarine target = event_target:system }
			transfer_ship = { prefer_name = "USS S22" type = submarine target = event_target:system }
			transfer_ship = { prefer_name = "USS S23" type = submarine target = event_target:system }

				annex_country = { target = event_target:system }
				add_manpower = 30000
