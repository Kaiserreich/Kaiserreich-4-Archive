		}
	}

			scoped_sound_effect = alert_enemy_generate_wargoal_sound
		}
		set_variable_to_random = global.kaiserreich_alerts_dirty
	}
}

state_event = { #event to fill the progress bar and remove the alert after three days (donotlog)
	id = generic_events.40
	hidden = yes

	trigger = {
		is_in_array = { ROOT.state_integration_alerts = event_target:integrated_state }
	}

	is_triggered_only = yes

	option = {
		set_temp_variable = { target = event_target:integrated_state }
		add_to_variable = { ROOT.integration_alert_timer_@var:target = 1 }
		if = {
			limit = { check_variable = { ROOT.integration_alert_timer_@var:target > 71 } }
			ROOT = { clear_integration_alert = yes }
		}
		else = {
			ROOT = { update_integration_progressbar = yes }
			set_variable_to_random = global.kaiserreich_alerts_dirty
			state_event = { id = generic_events.40 hours = 1 trigger_for = ROOT }
		}
	}
}

		every_other_country = {

						INT = { exists = yes }

				limit = { is_ally_with = INT }
				embargo_PREV = yes
			}
		}
		set_variable = { GER_faction = 1 }
	}
}

# There are five main puppet levels:
#  1. Integrated Puppet (integrated_puppet) - the lowest, and countries normally can't raise above this if set here
#  2. Occupied Puppet (occupied_puppet) - the default state, countries will be able to raise to the next level once they get some military, based on their size
#  3. Puppet (default_puppet) - the equivalent to the old colony level, only majors can generally raise higher than this on their own
#	4. Autonomous Puppet (autonomous_puppet) - the equivalent to the old dominion level, only more majors can raise higher than this on their own
#	5. Nominal Puppet (nominal_puppet) - a puppet in name only, barely restricted, majors can eventually move to freedom from this level
#
# NOTE: Currently there is no system in place for moving autonomy back downwards
#
