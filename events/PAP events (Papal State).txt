# The clearing of the Pontine Marshes is finished
country_event = {
	id = papal_state_events.6
	title = papal_state_events.6.t
	desc = papal_state_events.6.d
	picture = GFX_report_event_PAP_marshes

	fire_only_once = yes

	trigger = {
		OR = {
			tag = PAP
			tag = SIC
			tag = ITA
			tag = SRD
		}
		2 = { is_owned_and_controlled_by = ROOT }
		OR = {
			has_global_flag = { flag = PAP_pontine_marshes days > 365 }
			has_country_flag = PAP_pontine_marshes_cleared
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		log = "[GetLogInfo]: event papal_state_events.6 option papal_state_events.6.a"
		name = papal_state_events.6.a
		add_manpower = 30000
		if = {
			limit = { tag = PAP }
			add_stability = 0.10
		}
		else = {
			add_stability = 0.05
		}
		2 = {
			add_extra_state_shared_building_slots = 2
			add_building_construction = {
				type = industrial_complex
				level = 2
				instant_build = yes
			}
		}
		clr_country_flag = PAP_pontine_marshes_cleared
	}
}
