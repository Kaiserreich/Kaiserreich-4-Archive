country_event = { #Fate of Gabon
	id = annex.5000
	title = annexation_GAB_CONQ
	desc = annexation_GAB_CONQ_desc
	picture = GFX_report_event_generic_conference2

	is_triggered_only = yes

	immediate = {
		annexations_africa_create_french_event_target = yes
		if = {
			limit = {
				OR = {
					original_tag = MAF
					AND = {
						is_french_tag = yes
						has_socialist_government = no
					}
				}
			}
			save_event_target_as = can_annex_gabon
		}
	}

	option = { #liberate
		log = "[GetLogInfo]: event annex.5000 option annex.5000.a"
		name = annex.5000.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0
				has_event_target = can_annex_gabon
			}
			modifier = {
				factor = 0
				has_subject = MAF
			}
			modifier = {
				factor = 0
				is_major = no
				has_event_target = france
			}
		}
		set_temp_variable = { tag_to_release = GAB }
		release_targeted_tag = yes
		recheck_annexations = yes
	}

	option = { #give it to France
		log = "[GetLogInfo]: event annex.5000 option annex.africa.give_to_france"
		name = annex.africa.give_to_france
		trigger = {
			has_event_target = france
		}
		ai_chance = {
			base = 60
			modifier = {
				factor = 0
				has_subject = MAF
			}
			modifier = {
				factor = 0
				is_major = yes
				event_target:france = {
					is_subject = yes
					NOT = { is_subject_of = ROOT }
				}
			}
		}
		GAB = { transfer_african_territory_to_france = yes }
	}

	option = { #give to MAF
		log = "[GetLogInfo]: event annex.5000 option annex.africa.give_to_maf"
		name = annex.africa.give_to_maf
		trigger = {
			has_subject = MAF
		}
		ai_chance = { base = 100 }
		set_temp_variable = { tag_to_release = MAF }
		GAB = {
			every_core_state = { add_claim_of_target = yes }
			set_transfer_target_cores = yes
		}
		transfer_targeted_territory = yes
		MAF = { occupy_small_country = yes }
		recheck_annexations = yes
	}

	option = { #annex
		log = "[GetLogInfo]: event annex.5000 option annex.integration"
		name = annex.integration
		trigger = {
			has_event_target = can_annex_gabon
		}
		ai_chance = { base = 100 }
		GAB = { every_core_state = { add_claim_of_ROOT = yes } }
		occupy_small_country = yes
		if = {
			limit = { original_tag = NFA }
			NFA_Resistance_Policy_on_colony_states = yes
		}
		recheck_annexations = yes
	}

	option = { #occupy
		log = "[GetLogInfo]: event annex.5000 option annex.occupation"
		name = annex.occupation
		trigger = {
			NOT = { has_event_target = can_annex_gabon }
		}
		ai_chance = { base = 0 }
		set_country_flag = GAB_CONQ
		occupy_small_country = yes
		recheck_annexations = yes
	}
}
