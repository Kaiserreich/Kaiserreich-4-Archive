###########################
# City Capture Events
###########################
#
# If you want to add more city capture events:
#
# First add an entry for the city in KR_on_actions_city_capture in the common\on_actions folder
#
# Then add a news event here - you can copy worldnews.251 for the most basic version
# which provides a separate comment for the owner (FROM.FROM), the conqueror (FROM), the country receiving the event (THIS) and then everyone else
#

add_namespace = worldnews

# The Fall of London
news_event = {
	id = worldnews.251
	immediate = { log = "[GetLogInfo]: event worldnews.251" }
	title = worldnews.251.t
	desc = {
		text = worldnews.251.d.generic
		trigger = {
			NOT = {
				FROM = { tag = ENT }
				FROM = { tag = IRE }
				FROM.FROM = { tag = ENG }
			}
		}
	}
	desc = {
		text = worldnews.251.d.entente
		trigger = {
			FROM = { tag = ENT }
			FROM.FROM = { tag = ENG }
		}
	}
	desc = {
		text = worldnews.251.d.ireland
		trigger = {
			FROM = { tag = IRE }
			FROM.FROM = { tag = ENG }
		}
	}
	picture = GFX_news_event_capture_london

	major = yes
	is_triggered_only = yes

	trigger = {
		wants_world_news = yes
	}

	immediate = {
		event_target:fallen_state = { set_state_flag = city_has_fallen_once }
	}

	option = { #This is only a temporary setback!
		name = worldnews.251.a
		trigger = {
			tag = FROM.FROM
		}
	}

	option = { #This is the beginning of the end
		name = worldnews.251.b
		trigger = {
			tag = FROM
			NOT = { tag = IRE }
		}
	}

	option = { #With London in their hands, British defeat is inevitable
		name = worldnews.251.c
		trigger = {
			NOT = {
				tag = FROM.FROM
				tag = FROM
				AND = {
					OR = {
						FROM = { tag = ENT }
						FROM = { tag = IRE }
					}
					FROM.FROM = { tag = ENG }
				}
			}
		}
	}
	option = { #ROTK
		name = worldnews.251.e
		trigger = {
			NOT = {
				tag = FROM.FROM
				tag = FROM
			}
			FROM = { tag = ENT }
			FROM.FROM = { tag = ENG }
		}
	}
	option = { #Ireland
		name = worldnews.251.f
		trigger = {
			NOT = {
				tag = FROM.FROM
			}
			FROM = { tag = IRE }
			FROM.FROM = { tag = ENG }
		}
	}
}
