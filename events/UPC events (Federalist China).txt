country_event = { #Election Campaigning Begins
	id = UPC.federalelections.000
	title = UPC.federalelections.000.t
	desc = UPC.federalelections.000.d
	picture = GFX_report_event_CHN_china_five_colour_speech

	is_triggered_only = yes #Every time we want to start elections, need to fire this event.
	trigger = {
		OR = {
			AND = {
				has_country_flag = UPC_has_federalist_democracy_flag #Make sure there's no democratic backsliding
				has_country_flag = UPC_has_drafted_federalist_constitution_flag #Make sure you've united China and drafted a constitution.
				NOT = { has_country_flag = UPC_actively_holding_elections_flag } #In case there's already an election going on somehow (some shenanigans of you taking Beijing and starting GXC's chain, then clicking Unite China and drafting the constitution).
			}
			AND = {
				tag = GXC
				#Allows you first election go through
				NOT = {
					has_country_flag = UPC_GRA_won_first_elections_flag
					has_country_flag = UPC_CDL_won_first_elections_flag
				}
			}
		}
	}

	immediate = {
		hidden_effect = {
			set_country_flag = UPC_actively_holding_elections_flag
			set_variable = { UPC.campaign_actions_var = 5 } #For a total of 10 "free" campaign actions (5 from here, 5 from campaigning)
			set_variable = { UPC.campaign_timer_var = 6 } #Timer var is added at the end of the mission, so the first time it times out, it is "1".

			#Setup the Missions
			activate_mission = UPC_upcoming_federal_elections_timer

			#Setup Campaign Decisions
			for_each_scope_loop = {
				array = global.federal_elections_targets_array
				UPC = { activate_targeted_decision = { decision = UPC_campaign_in_state_decision target = PREV } }
			}

			#Set Up Elections in 4 Years assuming unification is completed and you're still a democracy.
			if = {
				limit = { has_country_flag = UPC_has_drafted_federalist_constitution_flag }
				country_event = { id = UPC.federalelections.000 days = 1460 } #Election Campaigning Begins
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event UPC.federalelections.000 option UPC.federalelections.000.a"
		name = UPC.federalelections.000.a #...momentum is on the side of the GRA!
		custom_effect_tooltip = UPC_will_campaign_as_GRA_tt
		if = {
			limit = { has_country_flag = UPC_is_campaigning_as_CDL }
			clr_country_flag = UPC_is_campaigning_as_CDL
		}
	}
	option = {
		log = "[GetLogInfo]: event UPC.federalelections.000 option UPC.federalelections.000.b"
		name = UPC.federalelections.000.b #...the CDL is gaining initiative!
		custom_effect_tooltip = UPC_will_campaign_as_CDL_tt
		if = {
			limit = { NOT = { has_country_flag = UPC_is_campaigning_as_CDL } }
			set_country_flag = UPC_is_campaigning_as_CDL
		}
	}
	after = {
		if = {
			limit = {
				is_power_balance_in_range = {
					id = UPC_power_balance
					range > UPC_bop_emerging_idealism
				}
			}
			if = {
				limit = { UPC_incumbent_is_GRA_trigger = yes }
				UPC_GRA_large_effect = yes
			}
			else = {
				UPC_CDL_large_effect = yes
			}
			custom_effect_tooltip = UPC_incumbent_advantage_tt
		}
		else = {
			if = {
				limit = { UPC_incumbent_is_GRA_trigger = yes }
				UPC_CDL_large_effect = yes
			}
			else = {
				UPC_GRA_large_effect = yes
			}
			custom_effect_tooltip = UPC_incumbent_disadvantage_tt
		}
	}
}
