# India wins the war
country_event = {
	id = delhi.5
	title = delhi.5.t
	desc = delhi.5.d
	picture = GFX_report_event_PRF_war

	fire_only_once = yes

	trigger = {
		tag = RAJ
		has_war_with = AFG
		has_global_flag = RAJ_afghan_war
		controls_state = 442 #Peshawar
		controls_state = 445 #Quetta
		surrender_progress < 0.05
		AFG = {
			owns_state = 442 #Peshawar
			owns_state = 445 #Quetta
		}
	}

	mean_time_to_happen = {
		days = 5
	}

	option = { # take cores
		log = "[GetLogInfo]: event delhi.5 option delhi.5.a"
		name = delhi.5.a
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				OR = {
					has_game_rule = { rule = AFG_war option = AFG_WAR_PUPPET_AMANULLAH_PATAUT }
					has_game_rule = { rule = AFG_war option = AFG_WAR_PUPPET_NADIR_PATAUT }
				}
			}
		}
		AFG = {
			remove_state_core = 442 #Peshawar
			remove_state_core = 445 #Quetta
			remove_state_claim = 444 #Baluchistan
			white_peace_with_ROOT = yes
		}
		RAJ = {
			transfer_state = 442 #Peshawar
			transfer_state = 445 #Quetta
		}
		hidden_effect = {
			AFG = {
				country_event = afg.1
				country_event = { id = afg.8 days = 7 random_days = 36 }
			}
		}
	}
	option = { # puppet
		log = "[GetLogInfo]: event delhi.5 option delhi.5.b"
		name = delhi.5.b
		trigger = {
			OR = {
				controls_state = 267 #Kabul
				AND = {
					is_ai = yes
					OR = {
						has_game_rule = { rule = AFG_war option = AFG_WAR_PUPPET_AMANULLAH_PATAUT }
						has_game_rule = { rule = AFG_war option = AFG_WAR_PUPPET_NADIR_PATAUT }
					}
				}
			}
		}
		ai_chance = {
			base = 50
			modifier = {
				factor = 0
				OR = {
					has_game_rule = { rule = AFG_war option = AFG_WAR_KING_AMANULLAH_PATAUT }
					has_game_rule = { rule = AFG_war option = AFG_WAR_KING_NADIR_PATAUT }
				}
			}
		}
		add_political_power = -100
		AFG = {
			remove_state_core = 442 #Peshawar
			remove_state_core = 445 #Quetta
			white_peace_with_ROOT = yes
		}
		RAJ = {
			transfer_state = 442 #Peshawar
			transfer_state = 445 #Quetta
			puppet = AFG
		}
		hidden_effect = {
			country_event = { id = delhi.6 days = 1 }
			AFG = {
				country_event = afg.1
			}
		}
	}
	option = { #conquest
		log = "[GetLogInfo]: event delhi.5 option delhi.5.c"
		name = delhi.5.c
		ai_chance = { base = 0 }
		add_political_power = -150
		add_stability = -0.05
	}
}

# The Burmese Question
country_event = {
	id = delhi.21
	title = delhi.21.t
	desc = delhi.21.d
	picture = GFX_report_event_BRM_army

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event delhi.21 option delhi.21.a"
		name = delhi.21.a #Restore an independent Burma
		trigger = {
			tag = RAJ
		}
		ai_chance = { base = 20 }
		add_political_power = 100
		every_owned_state = {
			limit = {
				is_claim_or_core_of_ROOT = no
				OR = {
					is_core_of = BRM
					state = 701
					state = 72
					state = 756
				}
			}
			BRM = {
				transfer_state = PREV
			}
		}
		set_temp_variable = { tag_to_release = BRM }
		puppet_released_tag = yes
		hidden_effect = {
			for_each_scope_loop = {
				array = BRM.core_states
				remove_core_of = HND
				remove_core_of = RAJ
				remove_core_of = PRF
				remove_claim_by = RAJ
				remove_claim_by = HND
				remove_claim_by = PRF
			}
			BRM = { copy_government_of_ROOT = yes }
		}
	}
	option = {
		log = "[GetLogInfo]: event delhi.21 option delhi.21.c"
		name = delhi.21.c #Restore the Konbaung dynasty!
		trigger = {
			tag = PRF
		}
		ai_chance = { base = 80 }
		add_political_power = 100
		every_owned_state = {
			limit = {
				is_claim_or_core_of_ROOT = no
				OR = {
					is_core_of = BRM
					state = 701
					state = 72
					state = 756
				}
			}
			BRM = {
				transfer_state = PREV
			}
		}
		set_temp_variable = { tag_to_release = BRM }

			}

			limit = {
				is_claim_or_core_of_ROOT = no

					is_core_of = BRM
					state = 701

				}
			}
