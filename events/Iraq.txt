	desc = irqrev.1.d

		has_war_with = EGY
	}

	fire_only_once = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				291 = {

				force_disable_resistance = { clear = yes }
			}
			IRQ = {
				transfer_state = 291

					name = "Kamil Shabib"

				}
				set_politics = {
					ruling_party = authoritarian_democrat
					last_election = "1935.10.11"
					election_frequency = 48
					elections_allowed = yes
				}
				remove_all_ministers = yes
				add_ideas = {
					IRQ_Rashid_Ali_al_Gaylani_hog_npo
					IRQ_Naji_al_Suwaydi_for_npo
					IRQ_Yunis_al_Saba'wi_eco_npo
					IRQ_Muhammad_Naji_Shawkat_sec_npo
				}
				set_popularities = {
					totalist = 0
					syndicalist = 0
					radical_socialist = 16
					social_democrat = 20
					social_liberal = 5
					market_liberal = 6
					social_conservative = 3
					authoritarian_democrat = 34
					paternal_autocrat = 15
					national_populist = 1
				}
				declare_war_on = {
					target = OTT
					type = annex_everything
				}
				add_ideas = ott_syrian_uprising
				load_oob = IRQ_revolt
			}
			EGY = {
				add_to_faction = IRQ
			}
		}
	}

	option = {
		name = irqrev.1.a
		every_owned_state = {
			limit = {
				is_core_of = IRQ
			}
			ott_increase_revolt_risk = yes
		}

		#Egypt
		custom_effect_tooltip = tooltip_white_line
		EGY = {
			add_ideas = ott_arab_revolt_support
		}
	}
}
