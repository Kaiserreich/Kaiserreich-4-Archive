# A monarchist conspiracy
country_event = {
	id = romania.7
	title = romania.7.t
	desc = romania.7.d

	trigger = {
		tag = ROM

		has_idea = ROM_royalist_threat
		date > 1937.1.1

		NOT = {
			country_exists = RRM
		}

		# No civil war if Romania is a puppet
		is_subject = no

		# Controls the base two provinces
		AND = {
			78 = { is_owned_and_controlled_by = ROM }
			79 = { is_owned_and_controlled_by = ROM }
		}

		# If Bucharest Circle was not immediately purged, or if the country is unstable
		OR = {
			has_country_flag = bucharest_circle
			has_political_power < -200
			has_idea = low_stability
			has_idea = very_low_stability
			has_idea = disaster
		}

		# Only if the Iron Guard hasn't been ousted
		OR = {
			has_government = national_populist
			has_government = paternal_autocrat
		}
	}

	fire_only_once = yes

	mean_time_to_happen = {
		days = 150

		modifier = {
			factor = 0.8

			has_country_flag = did_not_purge_the_circle
		}
	}

	immediate = {
		hidden_effect = {
			every_state = {
				limit = { is_core_of = ROM }
				add_core_of = RRM
			}
		}
	}

	option = {
		name = romania.7.a # The traitors!

		remove_ideas = ROM_royalist_threat

		add_political_power = -200

		RRM = {
			transfer_state = 78 # Bessarabia
			transfer_state = 79 # Moldova

			if = {
				limit = { 80 = { is_owned_and_controlled_by = ROM } }
				transfer_state = 80 # Bucovina
			}

			if = {
				limit = { 789 = { is_owned_and_controlled_by = ROM } }
				transfer_state = 789 # East Transylvania
			}

			if = {
				limit = { 76 = { is_owned_and_controlled_by = ROM } }
				transfer_state = 76 # North Transylvania
			}

			transfer_technology = yes

			hidden_effect = {
				set_capital = 79

				load_oob = RRM

				# Add chiefs if ROM has them (to keep it balanced)
				if = {
					limit = { ROM = { has_idea_with_trait = army_chief } }
					add_ideas = RRM_Constantin_Sanatescu_carm_ade
				}
				if = {
					limit = { ROM = { has_idea_with_trait = navy_chief } }
					add_ideas = RRM_Nicolae_Caranfil_cnav_sco
				}
				if = {
					limit = { ROM = { has_idea_with_trait = air_chief } }
					add_ideas = RRM_Gheorghe_Jienescu_cair_ade
				}
			}
		}

		hidden_effect = { country_event = { id = romania.8 days = 7 } }
	}

}

# Hidden event to dismantle the Serbian-Romanian Alliance
country_event = {
	id = romania.15

	hidden = yes

	trigger = {
		tag = SER
		is_subject = no

		has_global_flag = romanian_civil_war

		OR = {
			NOT = {
				ROM = {
					has_war_with = RRM
				}
				country_exists = RRM
			}
			NOT = {
				RRM = {
					has_war_with = ROM
				}
				country_exists = ROM
			}
		}

	OR = {
		is_in_faction_with = ROM
		is_in_faction_with = RRM
	}
	}

	mean_time_to_happen = {
		days = 1
	}

	fire_only_once = yes

	immediate = {
		if = {
			limit = { SER = { is_in_faction_with = ROM } }
			ROM = {
				dismantle_faction = yes
			}
		}
		if = {
			limit = { SER = { is_in_faction_with = RRM } }
			RRM = {
				dismantle_faction = yes
			}
		}
	}

	option = { name = romania.15.a }
}
