		text = mac.226.d.MON
		trigger = { NOT = { original_tag = TIB } }
	}
	desc = {
		text = mac.226.d.TIB
		trigger = { original_tag = TIB }
	}

	picture = GFX_report_event_TIB_german_mission_tibet

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event mac.226 option mac.226.a"
		name = mac.226.a #Yes, let's end this
		add_political_power = 50
		FROM = {
			white_peace_with_ROOT = yes
			hidden_effect = {
				if = {
					limit = {
						OR = {
							NOT = { has_war_with = MON }
							NOT = { has_war_with = TIB }
						}
					}
					remove_ideas = has_scripted_peace
				}
			}
			if = {
				limit = { ROOT = { original_tag = TIB } }
				### free MON if it was puppeted or annexed
				if = {
					limit = { has_subject = MON }
					set_autonomy = {
						target = MON
						autonomy_state = autonomy_free
					}
					every_owned_state = {
						limit = {
							OR = {
								region = 152 #Outer Mongolia
								region = 222 #Inner Mongolia
								state = 618 #Dzungaria
							}
						}
						transfer_state_to = MON
					}
					hidden_effect = {
						MON = { set_politics = { ruling_party = authoritarian_democrat } }
					}
				}
				else_if = {
					limit = {
						owns_state = 330
						NOT = { country_exists = MON }
					}
					effect_tooltip = { release = MON }
					hidden_effect = {
						every_owned_state = {
							limit = {
								OR = {
									region = 152 #Outer Mongolia
									region = 222 #Inner Mongolia
									state = 618 #Dzungaria
								}
							}
							transfer_state_to = MON
						}
						if = {
							limit = { MON = { has_government = paternal_autocrat } }
							MON = { set_politics = { ruling_party = authoritarian_democrat } }
						}
					}
				}
				### give MON its claims
				else = {
					every_owned_state = {
						limit = {
							OR = {
								region = 152 #Outer Mongolia
								region = 222 #Inner Mongolia
								state = 618 #Dzungaria
							}
						}
						transfer_state_to = MON
					}
				}

				### give TIB its claims
				every_owned_state = {
					limit = {
						OR = {
							region = 146 #Himalayas
							state = 353 #Yushu
							state = 604 #Xining
						}
					}
					transfer_state_to = ROOT
				}
				hidden_effect = {
					if = {
						limit = { MON = { is_ai = yes } } #As long as Mongolia is not controlled by a player, Tibet will control if there is a ceasefire.
						SHX = { country_event = { id = shx_suiyuan_campaign.22 days = 1 } }
					}
					else_if = { #If both Tibet and Mongolia are controlled by players, need to give a timer so they can't wait for the event to time out.
						limit = { TIB = { is_ai = no } }
						SHX = { country_event = { id = shx_suiyuan_campaign.22 days = 16 } } #If one makes peace and the other doesn't , this event won't fire.
					}
				}
			}
			else = {
				### free TIB if it was puppeted or annexed
				if = {
					limit = { has_subject = TIB }
					set_autonomy = {
						target = TIB
						autonomy_state = autonomy_free
					}
					every_owned_state = {
						limit = {
							OR = {
								region = 146 #Himalayas
								state = 353 #Yushu
								state = 604 #Xining
							}
						}
						transfer_state_to = TIB
					}
					hidden_effect = { TIB_remove_puppet_leader = yes }
				}
				else_if = {
					limit = {
						owns_state = 322
						NOT = { country_exists = TIB }
					}
					effect_tooltip = { release = TIB }
					hidden_effect = {
						every_owned_state = {
							limit = {
								OR = {
									region = 146 #Himalayas
									state = 353 #Yushu
									state = 604 #Xining
								}
							}
							transfer_state_to = TIB
						}
						TIB_remove_puppet_leader = yes
					}
				}
				### give TIB its claims
				else = {
					every_owned_state = {
						limit = {
							OR = {
								region = 146 #Himalayas
								state = 353 #Yushu
								state = 604 #Xining
							}
						}
						transfer_state_to = TIB
					}
				}

				### give MON its claims
				every_owned_state = {
					limit = {
						OR = {
							region = 152 #Outer Mongolia
							region = 222 #Inner Mongolia
							state = 618 #Dzungaria
						}
					}
					transfer_state_to = ROOT
				}
				hidden_effect = {
					if = {
						limit = { MON = { is_ai = no } } #If the player is Mongolia, then their decision to make peace will determine if war goes on
						SHX = { country_event = { id = shx_suiyuan_campaign.22 days = 1 } }
					}
				}
			}
		}
	}

		FROM = { country_event = mac.227 }
